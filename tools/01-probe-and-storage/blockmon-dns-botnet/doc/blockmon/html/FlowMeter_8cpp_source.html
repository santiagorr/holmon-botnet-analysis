33  * <blockinfo type="FlowMeter" invocation="indirect" thread_exclusive="False">
34  * <humandesc>
35  * Receives packet messages and keeps a table of per-flow statistics, and, depending on the configuration, also the packets making up that flow.
36  * A circular queue is used to keep track of the last time a flow has been checked for
37  * expiration.
38  * Two timeout values are used. The first one is the idle timeout (default value is 500 ms): whenever a flow record has not been
39  * receiving a packet for more than such time interval, the flow is considered as expired, its associated
40  * record is sent out and its table entry erased
41  *. The second one, the active timeout (default value is 100 ms), is activated when a flow has been active for longer than the corresponding time
42  * interval. When such timeout expires, a record about the flow is sent out and the counters in the corresponding table entry
43  * are reset.
44  * Notice that, while the idle timeout is checked by using a Blockmon timer mechanism, the indirect timeout is checked upon
45  * arrival of a new packet belonging to the flow.
46  * </humandesc>
47  * <shortdesc>Receives packet messages and keeps a table of per-flow statistics, and, depending on the configuration, also the packets making up that flow.</shortdesc>

123  static unsigned int simplehash(const char* b,size_t s)
124  {
125  int remaining_length=s;
126  unsigned int hash=0x66B566B5;
127  const char* it = b;
128  while(remaining_length >= 2)
129  {
130  hash ^= (hash << 7) ^ (*it++) * (hash >> 3);
131  hash ^= (~((hash << 11) + ((*it++) ^ (hash >> 5))));
132  remaining_length -= 2;
133  }
134  if (remaining_length)
135  {
136  hash ^= (hash << 7) ^ (*it) * (hash >> 3);
137  }
138  return hash;
139  }

148  {
149  FlowKey local;
150  memset(&local,0,sizeof(local)); //we need to 0 any random garbage in the padding
151  local = t;
152  return simplehash(reinterpret_cast<const char*> (&local),sizeof(FlowKey) );
153  }

295  void _handle_timer(std::shared_ptr<Timer>&& )
296  {
297  //flows whose last packet is older than this limit are considered as expired
298  ustime_t tnow = get_BM_time();
299  ustime_t us_limit = tnow - m_idle_timeout_ms * 1000;
300  while(1)
301  {
302  queue_slot cur_slot;
303  if(!pop_slot(cur_slot))
304  return; //queue is empty
305  const Flow* cur_flow = static_cast<const Flow*> (cur_slot.flow_ptr.get());
306  if(cur_flow->end_time() < us_limit) //timeout expired for this flow+
307  {
308  m_flow_table.erase(cur_flow->key());
309  printf("sending info out from handle_timer!\n");
310  send_out_through(std::move(cur_slot.flow_ptr),m_out_gate_id);
311  }
312  else
313  {
314  bool last_slot = (cur_slot.last_checked > us_limit);
315  cur_slot.last_checked = tnow;
316  push_slot(std::move(cur_slot)); //enqueue again
317  if(last_slot)
318  return;
319 
320  }
321  }

332  {
333  if(m->type()!=MSG_ID(Packet))

337  auto flow_it = m_flow_table.find( packet->key());
338  ustime_t us_now = get_BM_time();
339  if (flow_it == m_flow_table.end())
340  {

343  /*printf("insert flow: <%s:%d -> %s:%d proto=%d>\n",

345  new_flow->key().src_port,

350  new_flow->expand_interval (us_now);

352  new_flow->increment_packets (1);
353  if (m_store_packets)
354  new_flow->add_packet(std::dynamic_pointer_cast<const Packet>(m));

356  m_flow_table[packet->key()] = flow_sp;
357  queue_slot new_slot;

359  new_slot.last_checked = us_now;
360  if(!push_slot(std::move(new_slot)))
361  throw std::runtime_error("FlowMeter: queue is full");
362  }
363  else
364  {
365  Flow* flow = const_cast<Flow*> (static_cast<const Flow*>(flow_it->second.get()));
366  if( ((us_now - flow->start_time()) / 1000) > m_active_timeout_ms )
367  {
368  flow->expand_interval (us_now);
369 
370  /* printf("exporting flow: <%s:%d -> %s:%d proto=%d>\n",
371  ip_to_string(flow->key().src_ip4).c_str(),
372  flow->key().src_port,
373  ip_to_string(flow->key().dst_ip4).c_str(),
374  flow->key().dst_port,
375  flow->key().proto);*/
376 
377  send_out_through(flow_it->second->clone(),m_out_gate_id);
378  flow->reset();
379  }
380  flow->expand_interval (us_now);
381  flow->increment_bytes (packet->length());
382  flow->increment_packets (1);
383  if (m_store_packets)
384  flow->add_packet(std::dynamic_pointer_cast<const Packet>(m));

386  }
