32 /*
33  * <blockinfo type="FlowFilter" invocation="direct, indirect" thread_exclusive="False">
34  * <humandesc>
35  * </humandesc>
36  * Selects packets based on a set of given conditions. Selected packets are
37  * sent to the block's output gate, non-selected ones are dropped. A packet
38  * is selected if it matches *all* of the given conditions, and each of the
39  * conditions can be negated. Logically the selection resolves to
40  *
41  * select = [not]cond1=val AND [not]cond2=val AND [not]cond3=val ...
42  *
43  * where [not] means that the negation is optional. The conditions can be of
44  * the following kind:
45  *
46  * ip_src
47  * ip_dst
48  * l4_protocol
49  * src_port
50  * dst_port
51  *
52  * Please refer to the schema and examples below to see how to specify
53  * selection filters. Note that the negation (i.e., "not") is done via the
54  * "behavior" parameter, which can be set to "accept" or "discard" = negate.
55  * Default is "accept".
56  *
57  * <shortdesc>Selects packets based on conditions specified over the 5-tuple</shortdesc>
58  *
59  * <gates>
60  * <gate type="input" name="in_msg" msg_type="Msg" m_start="0" m_end="0" />
61  * </gates>
62  *
63  * <paramsschema>
64  * element params {
65  * element ip_src{
66  * attribute address {text}
67  * attribute netmask {text}
68  * element filter_mode{
69  * attribute behavior = {"discard"|"accept"}
70  * }?
71  * element ip_dst{
72  * attribute address {text}
73  * attribute netmask {text}
74  * element filter_mode{
75  * attribute behavior = {"discard"|"accept"}
76  * }?
77  * element l4_protocol{
78  * attribute number {xsd:integer}
79  * element filter_mode{
80  * attribute behavior = {"discard"|"accept"}
81  * }?
82  * element src_port{
83  * attribute number {xsd:integer}
84  * element filter_mode{
85  * attribute behavior = {"discard"|"accept"}
86  * }?
87  * element dst_port{
88  * attribute number {xsd:integer}
89  * element filter_mode{
90  * attribute behavior = {"discard"|"accept"}
91  * }?
92  * }
93  * }
94  * </paramsschema>
95  *
96  * <paramsexample>
97  * <params>
98  * <ip_src address="131.114.54.0" netmask="255.255.252.0">
99  * <filter_mode behavior="discard"/>
100  * </ip_src>
101  *
102  * <ip_dst address="131.114.54.0" netmask="255.255.252.0">
103  * <filter_mode behavior="accept"/>
104  * </ip_dst>
105  *
106  * <l4_protocol number="6">
107  * <filter_mode behavior="accept"/>
108  * </l4_protocol>
109  *
110  * <src_port number="80">
111  * <filter_mode behavior="accept"/>
112  * </src_port>
113  *
114  * </params>
115  * </paramsexample>
116  *
117  * <variables>
118  * </variables>
119  *
120  * </blockinfo>
121  */

181  static inline bool parse_filter_mode(pugi::xml_node& field_node, uint8_t& mode_byte)
182  {
183  pugi::xml_node mode;
184  if(!(mode = field_node.child("filter_mode")))
185  return false;
186  mode_byte |= 1<<FIELD_ACTIVE;
187  if(pugi::xml_attribute behavior = mode.attribute("behavior"))
188  {
189  if(!(strcmp(behavior.value(),"discard")))
190  mode_byte |= 1<<FILTER_OUT;
191  else if (strcmp(behavior.value(),"accept"))
192  return false;
193  }
194  else
195  return false;
196 
197  return true;
198  }

218  static inline bool go_ahead(bool filter_result, uint8_t filter_mode)
219  {
220  return (((filter_mode & (1<<FILTER_OUT)) && (!filter_result)) ||
221  ((!(filter_mode & (1<<FILTER_OUT))) && filter_result)) ;
222  }

374  virtual void _receive_msg(std::shared_ptr<const Msg>&& m, int /* index */)
375  {
376  if(m->type() != MSG_ID(Flow))
377  throw std::runtime_error("FlowFilter: wrong message type");
378  const Flow* flow = static_cast<const Flow* > (m.get());
379 
380  if(is_indirect(m_ip_src_mode))
381  {
382  if(!go_ahead( (m_ip_src_address & m_ip_src_mask) == (flow->key().src_ip4 & m_ip_src_mask), m_ip_src_mode))
383  return;
384  }
385 
386  if(is_indirect(m_ip_dst_mode))
387  {
388  if(!go_ahead( (m_ip_dst_address & m_ip_dst_mask) == (flow->key().dst_ip4 & m_ip_dst_mask), m_ip_dst_mode))
389  return;
390  }
391 
392  if(is_indirect(m_layer4_mode))
393  {
394  if(!go_ahead(m_layer4_proto == flow->key().proto, m_layer4_mode))
395  return;
396  }
397  if( (flow->key().proto != Packet::kUDP) && (flow->key().proto != Packet::kTCP) )
398  flow_passed(std::move(m));
399 
400 
401  if(is_indirect(m_src_port_mode))
402  {
403  if(!go_ahead(m_src_port == flow->key().src_port, m_src_port_mode))
404  return;
405  }
406 
407  if(is_indirect(m_dst_port_mode))
408  {
409  if(!go_ahead(m_dst_port == flow->key().dst_port, m_dst_port_mode))
410  return;
411  }
412 
413  flow_passed(std::move(m));
414 
415  }
