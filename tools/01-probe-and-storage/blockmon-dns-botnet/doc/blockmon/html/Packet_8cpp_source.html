43 /* Decode IP header HLV field into version and header length */
44 static size_t decode_hlv(uint8_t hlv, uint8_t& version) {
45  version = (hlv & 0xf0) >> 4;
46  return (hlv & 0x0f) << 2;
47 }

49 /* Decode TCP header payload offset field */
50 static size_t decode_payoff(uint8_t payoff) {
51  return (payoff & 0xf0) >> 2;
52 }

54 /* Parses MAC header, populates m_pkttype, m_ipbuffer if IP */
55 void Packet::parse_mac() const {
56  // shortcut valid ethertype
57  if (m_machdr_parsed) return;
58  m_machdr_parsed = true;
59 
60  const uint8_t *macbuf = m_buffer;
61 
62  if (m_mactype == kEthernet) {
63  // Shortcut further parsing on runt packet
64  if (m_caplen < sizeof(ethhdr)) {
65  m_pkttype = kPktTypeRuntL2;
66  m_iphdr_parsed = true;
67  m_ports_parsed = true;
68  m_l4hdr_parsed = true;
69  return;
70  }
71 
72  // get packet type
73  const ethhdr *eh = reinterpret_cast<const ethhdr*>(m_buffer);
74  m_pkttype = ntohs(eh->ethertype);
75  macbuf += sizeof(ethhdr);
76 
77  while (m_pkttype == kPktType1q) {
78  if ((m_caplen - (macbuf - m_buffer)) < sizeof(ethhdr)) {
79  m_pkttype = kPktTypeRuntL2;
80  m_iphdr_parsed = true;
81  m_ports_parsed = true;
82  m_l4hdr_parsed = true;
83  return;
84  }
85 
86  // 802.1q header; parse and try again
87  const eth1qhdr *e1qh = reinterpret_cast<const eth1qhdr*>(macbuf);
88  m_pkttype = ntohs(e1qh->ethertype);
89  macbuf += sizeof(eth1qhdr);
90  }
91  } else if (m_mactype == kRawIP) {
92  m_pkttype = kPktTypeIP4;
93  }
94 
95  if (m_pkttype == kPktTypeIP4) {
96  // Store pointer to IP header
97  m_iphdr_ptr = macbuf;
98  } else {
99  // Shortcut further parsing if not IP
100  m_iphdr_parsed = true;
101  m_ports_parsed = true;
102  m_l4hdr_parsed = true;
103  }
104 }

106 /* parses IP header, populates m_ip*, m_sip, m_dip, m_key.proto */
107 void Packet::parse_iphdr() const {
108  // shortcut valid iphdr
109  if (m_iphdr_parsed) return;
110  m_iphdr_parsed = true;
111 
112  // make sure we've parsed the mac header
113  parse_mac();
114 
115  // verify we have enough valid IP header
116  if (m_iphdr_ptr) {
117  size_t iphoff = m_iphdr_ptr - m_buffer;
118  if ((m_caplen - iphoff) >= sizeof(ip4hdr)) {
119  // Yep. Parse header.
120  const ip4hdr *iph = reinterpret_cast<const ip4hdr*>(m_iphdr_ptr);
121  m_key.src_ip4 = ntohl(iph->sip4);
122  m_key.dst_ip4 = ntohl(iph->dip4);
123  m_key.proto = iph->proto;
124  m_iptos = iph->tos;
125  m_ipttl = iph->ttl;
126  m_iplen = ntohs(iph->len);
127 
128  // get header length
129  uint8_t version;
130  m_iphlen = decode_hlv(iph->hlv, version);
131 
132  // find layer 4 header
133  if ((m_caplen - iphoff) >= m_iphlen) {
134  // we may even have a valid layer 4 header, store it
135  m_l4hdr_ptr = m_iphdr_ptr + m_iphlen;
136  }
137  } else {
138  // Shortcut further parsing on runt IP
139  m_pkttype = kPktTypeRuntL3;
140  m_ports_parsed = true;
141  m_l4hdr_parsed = true;
142  }
143  }
144 }

146 /* if protocol is UDP or TCP, populates m_key.src_port and m_key.dst_port;
147  also finds payload for UDP */
148 void Packet::parse_ports() const {
149  // shortcut valid ports
150  if (m_ports_parsed) return;
151  m_ports_parsed = true;
152 
153  // make sure we've parsed the IP header
154  parse_iphdr();
155 
156  // verify we have valid ports
157  if (m_l4hdr_ptr && ((m_key.proto == kTCP) || (m_key.proto == kUDP))) {
158  size_t l4hoff = m_l4hdr_ptr - m_buffer;
159  if ((m_caplen - l4hoff) >= sizeof(porthdr)) {
160  const porthdr *ph = reinterpret_cast<const porthdr*>(m_l4hdr_ptr);
161  m_key.src_port = ntohs(ph->sp);
162  m_key.dst_port = ntohs(ph->dp);
163 
164  // go ahead and skip the rest of the UDP header
165  if ((m_key.proto == kUDP) && ((m_caplen - l4hoff) >= sizeof(udphdr))) {
166  const udphdr* uh = reinterpret_cast<const udphdr*>(m_l4hdr_ptr);
167  m_udplen = ntohs(uh->len);
168  m_l4payload_ptr = m_l4hdr_ptr + sizeof(udphdr);
169  m_payload_len = m_iplen - m_iphlen - sizeof(udphdr);
170  }
171  } else {
172  // Shortcut further parsing on runt ports
173  m_pkttype = kPktTypeRuntL4;
174  m_l4hdr_parsed = true;
175  }
176  }
177 }

181 void Packet::parse_tcphdr() const {
182  // shortcut valid ports
183  if (m_l4hdr_parsed) return;
184  m_l4hdr_parsed = true;
185 
186  // insure IP parsed
187  parse_iphdr();
188 
189  // look for a valid TCP header
190  if (m_l4hdr_ptr && (m_key.proto == kTCP)) {
191  size_t tcphoff = m_l4hdr_ptr - m_buffer;
192  if ((m_caplen - tcphoff) >= sizeof(tcphdr)) {
193  // Yep. Parse header.
194  const tcphdr *tcph = reinterpret_cast<const tcphdr*>(m_l4hdr_ptr);
195  m_key.src_port = ntohs(tcph->sp);
196  m_key.dst_port = ntohs(tcph->dp);
197  m_tcpflags = tcph->flags;
198  m_tcpseq = ntohl(tcph->seqnum);
199  m_tcpack = ntohl(tcph->acknum);
200  m_tcpurg = tcph->urgent;
201  m_tcpwin = tcph->window;
202 
203  // get header length
204  m_tcphlen = decode_payoff(tcph->payoff);
205 
206  // FIXME parse options
207 
208  // find payload
209  if ((m_caplen - tcphoff) >= m_tcphlen) {
210  // we may even have a valid layer 4 payload, store it
211  m_l4payload_ptr = m_l4hdr_ptr + m_tcphlen;
212  m_payload_len = m_iplen - m_iphlen - m_tcphlen;
213  }
214  } else {
215  // Mark runt packet
216  m_pkttype = kPktTypeRuntL4;
217  }
218  }
