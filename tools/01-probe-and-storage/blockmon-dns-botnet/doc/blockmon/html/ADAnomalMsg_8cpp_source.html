109  vector<double> PDF_to_CCDF(vector<double> PDF)
110  {
111  vector<double> CDF, CCDF;
112  uint i;
113  for(i=0; i< PDF.size(); i++)
114  {
115  if(i==0)
116  CDF.push_back(PDF[i]);
117  else
118  CDF.push_back(CDF[i-1] + PDF[i]);
119  CCDF.push_back(1 - CDF[i]);
120  }
121  return CCDF;
122  }

125  std::string ADAnomalMsg::to_string() const
126  {
127  stringstream ss;
128  ss << "CCDF timestamp: " << m_timestamp
129  << " ; Timescale: " << m_timescale
130  << " ; Feature ID: " << m_featureID
131  << " ; CCDF_size: " << m_CCDF_size
132  << "\nCCDF: ";
133  vector<double>::const_iterator it;
134  for (it = m_CCDF.begin(); it != m_CCDF.end(); it++) {
135  ss << (*it) << " " ;
136  }
137  ss << "\nReference Set size: " << m_RefSet_size
138  << "\nReferenceSet CCDFs:\n";
139  window_element w_el;
140  window_array::const_iterator itt;
141  vector<double> RS_CCDF;
142  uint i;
143  for (itt = m_RefSet.begin(); itt != m_RefSet.end(); itt++) {
144  w_el = (*itt);
145  RS_CCDF = blockmon::PDF_to_CCDF(w_el.PDF);
146  for(i=0; i< RS_CCDF.size(); i++)
147  {
148  ss << RS_CCDF[i] << " ";
149  }
150  ss << "\n" ;
151  }
152  ss << "\n" ;
153  return ss.str();
154  }
