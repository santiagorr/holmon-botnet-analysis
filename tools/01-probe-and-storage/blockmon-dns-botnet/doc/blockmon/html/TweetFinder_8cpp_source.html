145  void _receive_msg(std::shared_ptr<const Msg>&& m, int /* index */)

151 
152  if(receivedTweets == 0) {

156  #ifdef _USE_JSMN_LIBRARY
157  const Tweet* msg_ptr = static_cast<const Tweet *>(m.get());
158 
159  // Parse the tweet
160  jsmn_init(&parser);
161  while(1) {
162  const char* buf = msg_ptr->get_tweet().c_str();
163  int r = jsmn_parse(&parser, buf, tokens, allocated_tokens);
164  if(r == JSMN_SUCCESS) {
165  // now loop to find the "text" part
166  // we check one token over two, for every JSMN_STRING check if it
167  // contains "text"
168  if(tokens[0].type != JSMN_OBJECT) {
169  ignored++;
170  //std::cerr << "No object found in tweet" << std::endl;
171  break;
172  }
173 
174  int size = tokens[0].size;
175 
176  int jj;
177  for(jj = 0; jj < size; jj += 2) {
178  if(tokens[1 + jj].type == JSMN_STRING && tokens[2 + jj].type == JSMN_STRING) {
179  if(!strncmp(buf + tokens[1 + jj].start, "text", 4)) {
180  tweet_found++;
181  char tmpbuf[16384];
182  int jcn;
183  for(jcn = tokens[2 + jj].start; jcn < tokens[2 + jj].end; jcn++) {
184  if(buf[jcn] == '#') {
185  break;
186  }
187  }
188  if(jcn < tokens[2 + jj].end) {
189  hashtag_found++;
190  int jcn_start = jcn;
191  jcn++;
192  int hashtag_len = 1;
193  while(jcn < tokens[2 + jj].end) {
194  if(!isalnum(buf[jcn]))
195  break;
196  jcn++;
197  hashtag_len++;
198  }
199  strncpy(tmpbuf, buf + jcn_start, hashtag_len);
200  tmpbuf[hashtag_len] = 0;
201  size_t val_hash = hash_fn((tmpbuf));
202  int ogate = val_hash % num_gates;
203  hashtag_batch[ogate]->add_word(std::string(tmpbuf));
204 #define BATH_SIZE 200
205  if(hashtag_batch[ogate]->get_word_number() >= BATH_SIZE) {
206  sentTuple++;
207  sentMessages += BATH_SIZE;
208  send_out_through(move(hashtag_batch[ogate]), m_outgate_ids[ogate]);
209  std::shared_ptr< WordRecord > m(new WordRecord());
210  hashtag_batch.at(ogate) = m;
211  }
212 
213  }
214  break;
215  }
216  }
217  }
218  break;
219  } else if(r == JSMN_ERROR_NOMEM) {
220  std::cerr << "Reallocating tokens" << std::endl;
221  allocated_tokens += ALLOC_TOKEN;
222  tokens = (jsmntok_t*) realloc(tokens, sizeof(jsmntok_t) * allocated_tokens);
223  if(tokens == NULL) {
224  throw std::runtime_error("Can not allocate memory\n");
225  }
226  continue;
227  } else if(r == JSMN_ERROR_INVAL) {
228  break;
229  } else if(r == JSMN_ERROR_PART) {
230  break;
231  } else {
232  throw std::runtime_error("Unknown error from jsmn\n");
233  }
234  }
235  #endif
236 
237  receivedTweets++;
238  m_msg_recv++;
239  }
