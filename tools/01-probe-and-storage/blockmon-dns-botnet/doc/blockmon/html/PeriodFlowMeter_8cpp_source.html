89  struct my_hash : public std::unary_function<FlowKey, size_t>
90  {
91  private:
97  static unsigned int simplehash(const char* b,size_t s)
98  {
99  int remaining_length=s;
100  unsigned int hash=0x66B566B5;
101  const char* it = b;
102  while(remaining_length >= 2)
103  {
104  hash ^= (hash << 7) ^ (*it++) * (hash >> 3);
105  hash ^= (~((hash << 11) + ((*it++) ^ (hash >> 5))));
106  remaining_length -= 2;
107  }
108  if (remaining_length)
109  {
110  hash ^= (hash << 7) ^ (*it) * (hash >> 3);
111  }
112  return hash;
113  }
114  public:
121  size_t operator()(const FlowKey& t) const
122  {
123  FlowKey local;
124  memset(&local,0,sizeof(local)); //we need to 0 any random garbage in the padding
125  local = t;
126  return simplehash(reinterpret_cast<const char*> (&local),sizeof(FlowKey) );
127  }
128  };

212  virtual void _receive_msg(std::shared_ptr<const Msg>&& m, int /* index */)
213  {
214  if(m->type()!=MSG_ID(Packet))
215  throw std::runtime_error("PeriodFlowMeter:: wrong message type");
216  const Packet* packet = static_cast<const Packet*> (m.get());
217  auto flow_it = m_flow_table.find( packet->key());
218  auto chrono_now = std::chrono::system_clock::now();
219  ustime_t us_now = (std::chrono::duration_cast<std::chrono::milliseconds> (chrono_now.time_since_epoch())).count();
220  if (flow_it == m_flow_table.end())
221  {
222  Flow* new_flow = new Flow (packet->key());
223  new_flow->expand_interval (us_now);
224  new_flow->increment_bytes (packet->length());
225  new_flow->increment_packets (1);
226  std::shared_ptr<const Msg> flow_sp (new_flow);
227  m_flow_table[packet->key()] = flow_sp;
228  }
229  else
230  {
231  Flow* flow = const_cast<Flow*> (static_cast<const Flow*>(flow_it->second.get()));
232  flow->expand_interval (us_now);
233  flow->increment_bytes (packet->length());
234  flow->increment_packets (1);
235  }
236  }
