381  uint32_t get_bin_for_count(uint64_t count, vector<uint64_t> b_ed)
382  {
383  /* We check if the zero filtering is activated and the counter is equal
384  * to 0 we have to discard this element returning an invalid
385  * index = 20*dim.
386  */
387  if(m_flag0) //it means that the zero filtering is activated
388  {
389  if(count == 0) //if the counter is zero we don't have to insert this element
390  {
391  return 20 * b_ed.size(); // we don't have to put this value into the array
392  }
393  }
394 
395  for (uint i=0; i<b_ed.size()-1; i++)
396  {
397  if( (count >= b_ed[i]) && (count < b_ed[i+1]) )
398  {
399  return i;
400  }else if ((i == b_ed.size()-2) && (count >= b_ed[i+1])) //just in case of counter > max_val!
401  {
402  return i;
403  }
404  }
405  return 10*b_ed.size();
406  }

444  void insert_elements(const TicketMsg* ticket)
445  {
446  const map<string, uint64_t>* ticket_features = ticket->get_features();
447  map<string, uint64_t>::const_iterator tck_feat_it;
448 
449  /* For each feature of the current user (ticket) */
450  for(tck_feat_it = ticket_features->begin(); tck_feat_it != ticket_features->end(); tck_feat_it ++)
451  {
452  /* Since we first fill in the histo hash table with dummy
453  * histogram entries and then we substitute the them with actual values,
454  * we should find for sure the histogram related to the feature we
455  * are interested in.
456  */
457  histo_hash_table::iterator histo_ht_it = m_histo_ht.find(tck_feat_it->first);
458  features_map::iterator feat_map_it = m_feat_info_map.find(tck_feat_it->first);
459 
460  /* If the feature is not in the histo_hash_table, it means that
461  * the user did not include this feature (and the related max_value)
462  * in the configuration file, so this feature has to be ignored! */
463  if( histo_ht_it == m_histo_ht.end()) continue;
464 
465  histo_entry* h_entry = &(histo_ht_it->second);
466  feat_info feat_entry = feat_map_it->second;
467 
468  //Given the current feature value, gets the index of the histogram element to increase.
469  uint32_t histo_index = get_bin_for_count(tck_feat_it->second, feat_entry.edges);
470  uint16_t nr_bins = feat_entry.histo_dim;
471 
472  /* Checks if the get_bin_for_count returned an invalid value,
473  * or it filtered out the current count value because of the
474  * zero filtering activated.
475  * In these cases we need to skip the current feature
476  * without inserting it in the hash_table, and to go
477  * to the next iteration (next feature in the current ticket)
478  * in the loop.
479  */
480  if(histo_index > nr_bins) continue;
481 
482  /* Increases the histogram element at the index position */
483  h_entry->histogram[histo_index] ++;
484 
485  /* Update the total value counter*/
486  h_entry->total_val += tck_feat_it->second;
487 
488  /* Updating the unique active users hash table */
489  uint32_t curr_user = ticket->get_userID();
490  map<uint32_t,uint32_t>::iterator users_it;
491  users_it = h_entry->unique_users_map.find(curr_user);
492  /*Only if the current user is not yet in the unique users hash table, we add it*/
493  if(users_it == h_entry->unique_users_map.end())
494  h_entry->unique_users_map.insert(pair<uint32_t,uint32_t>(curr_user, curr_user));
495  }
496 
497  return;
498  }

506  virtual void _receive_msg(std::shared_ptr<const Msg>&& m, int /* index */)
507  {
508  if(m->type() != MSG_ID(TicketMsg))
509  throw std::runtime_error("HistoCalc: wrong message type");
510  const TicketMsg* ticket = static_cast<const TicketMsg* > (m.get());
511 
512  uint64_t time_now = ticket->get_timestamp();
513 
514  /* First iteration -> initialize the hash table with dummy elements */
515  if(m_current_agg_time == 0)
516  {
517  insert_dummy_elements();
518  m_current_agg_time = time_now;
519  }
520 
521  while (time_now > m_current_agg_time)
522  {
523  /* For each element in histogram hash table, creates and sends
524  * a HistoMsg with timestamp m_current_time */
525  empty_hash_table(m_current_agg_time);
526 
527  /* Update current aggregate time */
528  m_current_agg_time += (m_timescale * 60);
529 
530  /* Initialize, with dummy elements, the histogram hash table
531  * entries for this time interval*/
532  insert_dummy_elements();
533  }
534 
535  /* In case of (dummy) signaling message, we understand that a new
536  * time interval has begun and we discard it!*/
537  if( (ticket->get_userID() == 0) && (ticket->get_extracted_features() == 0) )
538  return;
539 
540  /* Insert the information included in the current ticket message,
541  * in the histogram hash table*/
542  insert_elements(ticket);
543 
544  }
