160  virtual void _configure(const pugi::xml_node& n)
161  {
162  std::cout << __PRETTY_FUNCTION__ << std::endl;
163 
164  int offset = 0;
165  int slots = 131072;
166  int caplen = 1514;
167 
168  bool timestamp = false;
169 
170  pugi::xml_node queues = n.child("queues");
171  if(!queues)
172  throw std::runtime_error("PFQSource:: no queues node");
173 
174  m_device = std::string(queues.attribute("device").value());
175  if(m_device.length()==0)
176  throw std::runtime_error("PFQSource::no device attribute ");
177 
178  auto clattr = queues.attribute("caplen");
179  auto offattr = queues.attribute("offset");
180  auto slotattr = queues.attribute("slots");
181 
182  auto tsattr = queues.attribute("tstamp");
183 
184  if (!clattr.empty())
185  caplen = clattr.as_int();
186 
187  if (!offattr.empty())
188  offset = offattr.as_int();
189 
190  if (!slotattr.empty())
191  slots = slotattr.as_int();
192 
193  if (!tsattr.empty())
194  timestamp = tsattr.as_bool();
195 
196  std::vector<unsigned int> vq;
197 
198  m_pfq.open(caplen, offset, slots);
199 
200  for (xml_node queue=queues.child("queue"); queue; queue=queue.next_sibling("queue"))
201  vq.push_back(queue.attribute("number").as_uint());
202 
203  if(!vq.empty())
204  {
205  auto it = vq.begin();
206  auto it_e = vq.end();
207  for (;it!=it_e;++it)
208  {
209  m_pfq.add_device(m_device.c_str(),*it);
210  }
211  }
212  else
213  {
214  blocklog("PFQSource: no queues specified, sniffing on device ", log_info);
215  m_pfq.add_device(m_device.c_str(), net::pfq::any_queue);
216  }
217 
218  std::cout << "PFQ: dev:" << m_device << " caplen:" << caplen << " tstamp:" << std::boolalpha << timestamp;
219  std::cout << " queues:";
220  std::copy(vq.begin(), vq.end(), std::ostream_iterator<int>(std::cout, ","));
221  std::cout << std::endl;
222 
223  {
224  std::lock_guard<std::mutex> lock(m_mutex);
225  m_max_caplen = std::max(m_max_caplen, caplen);
226  }
227 
228  net::payload::size_of(m_max_caplen);
229 
230  // the slice_allocator must be created once the dynamic_buffer (or net::payload)

251  void _do_async()
252  {
253 
254  net::queue b = m_pfq.read(10 /* microseconds */);
255 
256  auto it_e = b.end();
257  for (auto it = b.begin(); it != it_e; ++it)
258  {
259  auto actual_payload = reinterpret_cast<uint8_t *>(it.data());
260 
261  while (!it->commit)
262  {}
263 
264 #if defined(USE_SIMPLE_PACKET) || defined(USE_SLICED_PACKET)
265  uint16_t len = it->len, caplen = it->caplen;
266  uint32_t sec = it->tstamp.tv.sec, nsec = it->tstamp.tv.nsec;
267 #endif
268 #if defined(USE_SIMPLE_PACKET)
269  auto sp = blockmon::NewPacket::SimplePacket(const_buffer<uint8_t>(actual_payload, it->caplen), *m_allocator, len, caplen, sec, nsec);
270 #elif defined(USE_SLICED_PACKET)
271  auto sp = blockmon::NewPacket::SlicedPacket(const_buffer<uint8_t>(actual_payload, it->caplen), *m_allocator, len, caplen, sec, nsec);
272 #else
273  auto sp = alloc_msg_from_buffer<Packet>(m_mem_block, it->caplen, const_buffer<uint8_t>(actual_payload, it->caplen));
274 #endif
275  send_out_through(std::move(sp), m_gate_id);
276  }
