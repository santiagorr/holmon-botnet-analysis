33  * <blockinfo type="CDFGenerator" invocation="both" thread_exclusive="false">
34  * <humandesc>
35  * Implements a block that prints the Cumulative Distribution Function
36  * (CDF) of the packet size every interval, adding a noise depending on
37  * the epsilon value. Configuration parameters allow to tune the
38  * endpoints of the CDF, the bin width and to add noise.
39  * </humandesc>
40  *
41  * <shortdesc>Prints the CDF of packet sizes for a given interval</shortdesc>
42  *
43  * <gates>
44  * <gate type="input" name="in_pkt" msg_type="Packet" m_start="0" m_end="0" />
45  * </gates>
46  *
47  * <paramsschema>
48  * element params {
49  * element cdf_build {
50  * attribute interval {xsd:integer}
51  * }
52  * element cdf_param{
53  * attribute min {xsd:integer}
54  * attribute max {xsd:integer}
55  * attribute bin {xsd:integer}
56  * }
57  * element diff_priv{
58  * attribute epsilon {xsd:float}
59  * }
60  * }

62  * </paramsschema>
63  *
64  * <paramsexample>
65  * <params>
66  * <build_cdf interval="2000"/> #millisecond
67  * <cdf_param min="0" max="1500" bin="150"/>
68  * <diff_priv epsilon="0.8"/> #if -1 do not add noise

72  * <variables>
73  * </variables>

75  * </blockinfo>

110  void CDFGenerator::_receive_msg(std::shared_ptr<const Msg>&& m, int /* index */){
111  if(m->type() != MSG_ID(Packet)) throw std::runtime_error("CDFGenerator: unexpected msg type");
112  const Packet* packet = static_cast<const Packet*>(m.get());
113 
114  //update histogram with the received value
115  int pkt_size = packet->length();
116  if (pkt_size >= m_min && pkt_size <= m_max)
117  m_histo[(pkt_size - m_min) / m_bin]++;
118  }

121  //extract a sample from a Laplace distribution
122  static double nextLaplace(double mu, double stddev){
123  double num = rand();
124  num /= RAND_MAX;
125 
126  if (num <= 0.5)
127  num = log ( 2 * num );
128  else
129  num = -log ( 2 * (1-num) );
130 
131  return stddev * num + mu;
132  }

134  //build differentially private noisy cdf
135  static void cdf_computation(std::vector<int>& histo, std::vector<int>& cdf, double epsilon){
136  unsigned int i;
137 
138  for (i=0; i<histo.size(); i++){
139 
140  if (i == 0)
141  cdf[i] = histo[i];
142  else
143  cdf[i] = cdf[i-1] + histo[i];
144 
145  if (epsilon > 0) //add noise
146  cdf[i] += round(nextLaplace(0,sqrt(2)/epsilon));
147  }
