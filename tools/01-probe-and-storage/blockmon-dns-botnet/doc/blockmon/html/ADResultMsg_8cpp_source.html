151  std::string ADResultMsg::to_string() const
152  {
153  stringstream ss;
154  ss << "CCDF timestamp: " << m_timestamp
155  << " ; Timescale: " << m_timescale
156  << " ; Feature ID: " << m_featureID
157  << " ; Number of unique users: " << m_nr_unique_users
158  << " ; Total value: " << m_total_value
159  << " ; CCDF_size: " << m_CCDF_size
160  << "\nCCDF:\n";
161  vector<double>::const_iterator it;
162  for (it = m_CCDF.begin(); it != m_CCDF.end(); it++) {
163  ss << (*it) << " " ;
164  }
165  ss << "\n";
166 
167  string phase;
168  if(m_phase) phase = "Learning";
169  else phase = "Detection";
170 
171  ss << "phase: " << phase
172  << " ; Acceptance Region Lower Bound: " << m_AR_LB
173  << " ; Acceptance Region Upper Bound: " << m_AR_UB
174  << " ; Internal Dispersion Mean: " << m_ID_mean
175  << " ; External Dispersion Mean: " << m_ED_mean
176  << " ; Anomaly code: " << m_Anomaly_code << "\n";
177  return ss.str();
178  }
