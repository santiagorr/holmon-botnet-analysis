49  /* Recursive read of DNS name following the rules of DNS name compression */
50  std::string* DNSPacket::read_dns_name(uint16_t* consumed, const uint8_t * data, const uint8_t * dns_start) {
51 
52  unsigned int i = 0;
53  uint8_t j = 0;
54  std::string *tmp;
55  std::string *ret = new std::string();
56  uint16_t fake = 0;
57  uint16_t jump;
58 
59  if (*data < 0xC0) { // check if the DNS name is stored here or linked somewhere else in packet
60  ret = new std::string();
61  j = *(data + i);
62  while (j != 0 && j < 0xC0) {
63  i++;
64  if (i != 1)
65  *ret += "."; // put dot before each domain except the first one
66  tmp = new std::string((const char*)(data + i) , j); // parse j symbols
67  *ret += *tmp;
68  i += tmp->length();
69  j = *(data + i); // load new start of domain
70  }
71  if (j != 0) {
72  jump = load_BE_u16(data + i) & 0x3FFF;
73  *consumed += i + 2; // increase number of consumed bytes
74  // we are no longer consuming packet bytes, so &fake is used
75  tmp = (read_dns_name(&fake, dns_start + jump, dns_start));
76 // std::cout<< "jump " << jump << " tmp " << *tmp << std::endl;
77  *ret += ".";
78  *ret += *tmp;
79  return (ret);
80  }
81  else
82  *consumed += i + 1; // consume the ending zero
83  }
84  else { // we hit the link so jump somewher else in the packet
85  jump = (load_BE_u16(data + i) & 0x3FFF);
86  ret = read_dns_name(&fake, dns_start + jump, dns_start);
87  *consumed += 2;
88  }
89 
90  return ret;
91 
92  }

94  /* Parsing of the DNS fields. Header first and then queries and answers. Authoritative and Additional sections
95  * are not being parsed at the moment.
96  */
97  int DNSPacket::parse_dns()
98  {
99  uint8_t dns_offset = payload() - m_buffer;
100 
101  transaction_id = load_BE_u16(m_buffer + dns_offset);
102 
103  QR_flag = *(m_buffer + dns_offset + 2) >> 0x7;
104  op_code = (*(m_buffer + dns_offset + 2) >> 3) & 0xF;
105  AA_flag = (*(m_buffer + dns_offset + 2) >> 2) & 0x1;
106  TC_flag = (*(m_buffer + dns_offset + 2) >> 1) & 0x1;
107  RD_flag = (*(m_buffer + dns_offset + 2) & 0x1);
108  RA_flag = (*(m_buffer + dns_offset + 3) >> 0x7);
109  AD_flag = (*(m_buffer + dns_offset + 3) >> 0x5) & 0x1;
110  ND_flag = (*(m_buffer + dns_offset + 3) >> 0x4) & 0x1;
111  re_code = *(m_buffer + dns_offset + 3) & 0xF;
112 
113  n_questions = load_BE_u16(m_buffer + dns_offset + 4);
114  n_answer = load_BE_u16(m_buffer + dns_offset + 6);
115  n_ns = load_BE_u16(m_buffer + dns_offset + 8);
116  n_additional = load_BE_u16(m_buffer + dns_offset + 10);
117 
118  struct q_record * q_r;
119 
120  uint16_t q_offset = dns_offset + 12;
121  uint16_t q_size = 0;
122  uint16_t consumed = 0;
123 
124  for (int i = 0; i < n_questions; i++) { // parse questions
125  q_r = new struct q_record;
126  consumed = 0;
127  q_r->qname = read_dns_name(&consumed, m_buffer + q_offset + q_size, m_buffer + dns_offset);
128  q_size += consumed; //initial length + ending zero
129  q_r->rtype = load_BE_u16(m_buffer + q_offset + q_size);
130  q_size += 2; // rtype
131  q_r->dclass = load_BE_u16(m_buffer + q_offset + q_size);
132  q_size += 2; // class
133  queries.push_back(*q_r);
134  }
135 
136  uint16_t a_offset = q_offset + q_size;
137  uint16_t a_size = 0;
138  uint16_t type;
139  uint16_t dclass;
140  uint32_t ttl;
141 // uint16_t len;
142  std::string * tmp;
143 
144  struct cname_record* cn_r;
145  struct a_record* a_r;
146 
147  for (int i = 0; i < n_answer; i++) { // parse answers
148  consumed = 0;
149  tmp = read_dns_name(&consumed, m_buffer + a_offset + a_size, m_buffer + dns_offset);
150  a_size += consumed;
151  type = load_BE_u16(m_buffer + a_offset + a_size);
152  a_size += 2;
153  dclass = load_BE_u16(m_buffer + a_offset + a_size);
154  a_size += 2;
155  ttl = load_BE_u32(m_buffer + a_offset + a_size);
156  a_size += 4;
157 // len = load_BE_u16(m_buffer + a_offset + a_size);
158  a_size += 2;
159 
160  switch (type) {
161  case 1: // RR type address
162  a_r = new struct a_record;
163  a_r->ttl = ttl;
164  a_r->rtype = type;
165  a_r->dclass = dclass;
166  a_r->qname = tmp;
167  a_r->ip_addr = load_BE_u32(m_buffer + a_offset + a_size);
168  a_size += 4;
169 
170  a_recs.push_back(*a_r);
171  break;
172  case 5: // RR type canonical name
173  cn_r = new struct cname_record;
174  cn_r->ttl = ttl;
175  cn_r->rtype = type;
176  cn_r->dclass = dclass;
177  cn_r->qname = tmp;
178  consumed = 0;
179  cn_r->cname = read_dns_name(&consumed, m_buffer + a_offset + a_size, m_buffer + dns_offset);
180  a_size += consumed;
181  c_names.push_back(*cn_r);
182  break;
183 
184  default:
185  break;
186  }
187  }
188 
189  return 0;
190  }

224  std::cout << "Operation code \t\t" << (int) op_code << "\t";
225  std::cout << "Authoritative answer \t" << (int) AA_flag << std::endl;
226  std::cout << "Truncated message \t" << (int) TC_flag << "\t";
227  std::cout << "Recursion desired \t" << (int) RD_flag << std::endl;
228  std::cout << "Recursion available \t" << (int) RA_flag << "\t";
229  std::cout << "Answer authenticated \t" << (int) AD_flag << std::endl;
230  std::cout << "Non-authenticated data\t" << (int) ND_flag << "\t";
231  std::cout << "Reply code\t\t" << (int) re_code << std::endl;
232 
233  std::cout << "==== DNS Sections ==== " << std::endl;
234  std::cout << "Questions " << n_questions << "\t\t\tAnswers " << n_answer << std::endl;
235  std::cout << "Authority " << n_ns << "\t\t\tAdditional " << n_additional << std::endl;
236 
237  std::cout << "==== DNS Questions ==== " << std::endl;
238  for (unsigned int i = 0; i < queries.size(); i++)
239  print_DNS_question(queries[i]);
240 
241  std::cout << "==== DNS Answers ==== " << std::endl;
242  for (unsigned int i = 0; i < c_names.size(); i++)
243  print_DNS_cname(c_names[i]);
244  for (unsigned int i = 0; i < a_recs.size(); i++)
245  print_DNS_arec(a_recs[i]);
246 
247  std::cout << std::endl;
248  }
