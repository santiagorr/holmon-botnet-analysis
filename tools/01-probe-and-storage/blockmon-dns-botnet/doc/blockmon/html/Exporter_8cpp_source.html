77 void Exporter::setTemplate(uint16_t tid)
78 {
79  // short circuit on already set
80  if (set_id_ == tid) {
81  return;
82  }
83 
84  // die on bad template ID
85  assert (tid >= kMinSetID);
86 
87  // Make sure template is already active
88  WireTemplate *new_tmpl = session_.getTemplate(domain_, tid);
89  assert (new_tmpl->isActive());
90 
91  // Finalize existing set if necessary
92  if (set_id_ && set_active_) endSet();
93 
94  // Set new template and set ID
95  tmpl_ = new_tmpl;
96  set_id_ = new_tmpl->tid();
97 
98  // create a new cursor
99  oc_ = ExporterOffsetCache(tmpl_, &xcoder_);
100 }

102 void Exporter::exportTemplatesForDomain() {
103  ensureTemplateSet();
104 
105  std::list<const WireTemplate *> tmpl_list = session_.activeTemplates(domain_);
106  for (auto iter = tmpl_list.begin(); iter != tmpl_list.end(); iter++) {
107  if ((*iter)->encodeTemplateRecord(xcoder_)) {
108  ++trec_count_;
109  } else {
110  flush();
111  ensureSet();
112  if ((*iter)->encodeTemplateRecord(xcoder_)) {
113  ++trec_count_;
114  } else {
115  // If it doesn't work now, it never will. Throw.
116  throw MTUError("template record");
117  }
118  }
119  }
120 
121  if (fast_flush_) flush();
122 }

125  ensureSet();

152 void Exporter::beginRecord() {
153  assert(!rec_active_);
154 
155  // Write a set header if we don't have one yet.
156  ensureSet();
157 
158  // Start a new message if we don't have at least enough
159  // space for a record.
160  checkRecordOverflow(tmpl_->minlen());
161  if (!tmpl_->varlenCount()) rec_will_fit_ = true;
162 
163  // Zero record
164  xcoder_.encodeZeroAt(tmpl_->minlen(), 0);
165 
166  // Create new record state
167  rec_active_ = true;
168 
169 // std::cerr << "----- begin record" << std::endl;
170 }

172 void Exporter::endRecord(bool do_export) {
173  assert(rec_active_);
174 
175  // Advance cursor to end of record if exporting
176  if (do_export) {
177  oc_.advance();
178  } else {
179  oc_.clear();
180  }
181 
182  // Clear record state
183  rec_active_ = false;
184  rec_will_fit_ = false;
185 
186  if (do_export) {
187  ++drec_count_;
188  if (fast_flush_) flush();
189  }
190 
191 // std::cerr << "----- end record" << std::endl;

208 bool Exporter::putValue(const InfoElement* ie, const void* vp, size_t len) {
209  // make sure the record's size is known and correct
210  assert(rec_will_fit_);
211 
212  // if the current template doesn't contain the value,
213  // just pretend we put it.
214  if (!tmpl_->contains(ie)) {
215  return true;
216  }
217 
218  // Figure out where to put the value.
219  size_t off = oc_.offsetOf(ie);
220 
221  // now encode, catch error
222  if (!xcoder_.encodeAt(vp, len, off, tmpl_->ieFor(ie))) {
223  return false;
224  }
225 
226  // Make sure the length is reserved
227  oc_.setLength(ie, len);
228 
229  return true;
230 }

242 void Exporter::ensureSet() {

266 void Exporter::flush(time_t export_time) {
267  //std::cerr << "flush(" << export_time << ")" << std::endl;
268 
269  if (msgEmpty()) {
270  return;
271  }
272 
273  if (set_active_) {
274  endSet();
275  }
276 
277  if (xcoder_.len() > kMessageHeaderLen) {
278  xcoder_.encodeMessageEnd(export_time, session_.incrementSequence(domain_, 0, drec_count_), domain_);
279  _sendMessage(buf_, xcoder_.len());
280  startMessage();
281  }
