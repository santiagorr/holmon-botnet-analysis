229  static inline bool parse_feature_flags(pugi::xml_node& feature_node, uint8_t& flags_byte)
230  {
231  pugi::xml_node flags_node;
232  //Check whether the field "flags" does exist, otherwise returns false
233  if(!(flags_node = feature_node.child("flags")))
234  return false;
235 
236  flags_byte = FILTER_OUT; //Inizialization -> no flag is set!(Flags = 0)
237 
238  //If the attribute "name" does exist
239  if(pugi::xml_attribute flags_name = flags_node.attribute("name"))
240  {
241  if (!strcmp(flags_name.value(),"_"))
242  {
243  flags_byte = 0;
244  }else if (!strcmp(flags_name.value(),"FIN"))
245  {
246  flags_byte = FIN;
247  }else if (!strcmp(flags_name.value(),"SYN"))
248  {
249  flags_byte = SYN;
250  }else if (!strcmp(flags_name.value(),"RST"))
251  {
252  flags_byte = RST;
253  }else if (!strcmp(flags_name.value(),"PSH"))
254  {
255  flags_byte = PSH;
256  }else if (!strcmp(flags_name.value(),"ACK"))
257  {
258  flags_byte = ACK;
259  }else if (!strcmp(flags_name.value(),"FINACK"))
260  {
261  flags_byte = FIN | ACK;
262  }else if (!strcmp(flags_name.value(),"SYNACK"))
263  {
264  flags_byte = SYN | ACK;
265  }else if (!strcmp(flags_name.value(),"RSTACK"))
266  {
267  flags_byte = RST | ACK;
268  }else if (!strcmp(flags_name.value(),"PSHACK"))
269  {
270  flags_byte = PSH | ACK;
271  }else if (!strcmp(flags_name.value(),"PSHFINACK"))
272  {
273  flags_byte = PSH | FIN | ACK;
274  }
275  }else return false; //Otherwise return false
276 
277  return true;
278  }

335  static inline bool parse_feature_mode(pugi::xml_node& feature_node, uint8_t& mode_byte, uint8_t direction)
336  {
337  pugi::xml_node mode_node;
338  //Check whether the field "feature mode" does exist, otherwise returns false
339  if(!(mode_node = feature_node.child("mode")))
340  return false;
341 
342  mode_byte = PACKETS_ACTIVE; //Inizialization = only "packets" is active
343 
344  //If the attribute "packets" does exist
345  if(pugi::xml_attribute packets = mode_node.attribute("packets"))
346  {
347  if(!strcmp(packets.value(),"on")) //strcmp returns 0 (false) if the two strings are equal
348  mode_byte |= PACKETS_ACTIVE;
349  }
350 
351  //If the attribute "bytes" does exist
352  if(pugi::xml_attribute bytes = mode_node.attribute("bytes"))
353  {
354  if(!strcmp(bytes.value(),"on")) //strcmp returns 0 (false) if the two strings are equal
355  mode_byte |= BYTES_ACTIVE;
356  }
357 
358  //only in uplink we can count the number of server ports and server IPs
359  if(direction == UPLINK)
360  {
361  if(pugi::xml_attribute sPorts = mode_node.attribute("ServPorts"))
362  {
363  if(!strcmp(sPorts.value(),"on"))
364  mode_byte |= NUMPORTS_ACTIVE;
365  }
366  if(pugi::xml_attribute sIPs = mode_node.attribute("ServIPs"))
367  {
368  if(!strcmp(sIPs.value(),"on"))
369  mode_byte |= NUMIPS_ACTIVE;
370  }
371  }
372 
373  return true;
374  }

405  static bool generate_feature_ID(const Packet *packet, string dirStr, uint16_t L4_prot, uint8_t flagsByte, string modeStr, uint16_t port, string& featIDStr)
406  {
407  /* UDP Datagram */
408  if(L4_prot & UDP)
409  {
410  featIDStr = featID_to_string(dirStr,"UDP",modeStr,port);
411  return true;
412  }
413 
414  /* No TCP flag set */
415  if((L4_prot & TCP) && (flagsByte == 0))
416  {
417  featIDStr = featID_to_string(dirStr,"TCP", modeStr, port);
418  return true;
419  }
420 
421  /* With some TCP flag set */
422  uint32_t ack; //useless
423  if(flagsByte & PSH)
424  {
425  if(flagsByte & ACK)
426  {
427  if(flagsByte & FIN)
428  {
429  //PSHFINACK
430  if(packet->tcp_has_fin() && packet->tcp_has_psh() && packet->tcp_ack(ack))
431  {
432  featIDStr = featID_to_string(dirStr,"PSHFINACK", modeStr, port);
433  return true;
434  }
435  //PSHACK
436  }else if(packet->tcp_ack(ack) && packet->tcp_has_psh())
437  {
438  featIDStr = featID_to_string(dirStr,"PSHACK", modeStr, port);
439  return true;
440  }
441  //PSH
442  }else if(packet->tcp_has_psh())
443  {
444  featIDStr = featID_to_string(dirStr,"PSH", modeStr, port);
445  return true;
446  }
447  }else if(flagsByte & RST)
448  {
449  if(flagsByte & ACK)
450  {
451  //RSTACK
452  if (packet->tcp_ack(ack) && packet->tcp_has_rst())
453  {
454  featIDStr = featID_to_string(dirStr,"RSTACK", modeStr, port);
455  return true;
456  }
457  //RST
458  }else if (packet->tcp_has_rst())
459  {
460  featIDStr = featID_to_string(dirStr,"RST", modeStr, port);
461  return true;
462  }
463  }else if(flagsByte & SYN)
464  {
465  if(flagsByte & ACK)
466  {
467  //SYNACK
468  if (packet->tcp_ack(ack) && packet->tcp_has_syn())
469  {
470  featIDStr = featID_to_string(dirStr,"SYNACK", modeStr, port);
471  return true;
472  }
473  //SYN
474  }else if (packet->tcp_has_syn())
475  {
476  featIDStr = featID_to_string(dirStr,"SYN", modeStr, port);
477  return true;
478  }
479  }else if(flagsByte & FIN)
480  {
481  if(flagsByte & ACK)
482  {
483  //FINACK
484  if (packet->tcp_ack(ack) && packet->tcp_has_fin())
485  {
486  featIDStr = featID_to_string(dirStr,"FINACK", modeStr, port);
487  return true;
488  }
489  //FIN
490  }else if (packet->tcp_has_fin())
491  {
492  featIDStr = featID_to_string(dirStr,"FIN", modeStr, port);
493  return true;
494  }
495  }else if(flagsByte & ACK)
496  {
497  //ACK
498  if (packet->tcp_ack(ack))
499  {
500  featIDStr = featID_to_string(dirStr,"ACK", modeStr, port);
501  return true;
502  }
503  }else return false;
504 
505  return false;
506  }

603  virtual void _receive_msg(std::shared_ptr<const Msg>&& m, int /* index */)
604  {
605  if(m->type() != MSG_ID(Packet))
606  throw std::runtime_error("FeaturesExtractor: wrong message type");
607  const Packet* packet = static_cast<const Packet* > (m.get());
608 
609  /*
610  * INSERIRE QUI IL CONTROLLO SUL TIMESTAMP DEL PACCHETTO RICEVUTO
611  * GENERARE E INVIARE IN OUTPUT UN MESSAGGIO DI SEGNALAZIONE (SEMPRE DI TIPO TICKETMSG)
612  * QUANDO TERMINA UN MINUTO E COMINCIA UN ALTRO IN MODO CHE I BLOCCHI
613  * SUCCESSIVI POSSANO "CAPIRE" CHE IL MINUTO CORRENTE É FINITO ED
614  * ESEGUIRE IMMEDIATAMENTE LE OPERAZIONI SUL MINUTO
615  *
616  * FAR PROPAGARE QUESTO MESSAGGIO ANCHE TRA I BLOCCHI SUCCESSIVI
617  * PER EVITARE L'ACCUMULO DI RITARDI TRA SCALE TEMPORALI
618  */
619  uint64_t time_now = packet->timestamp_s();
620  uint64_t minute_now = time_now/60;
621 
622  /*If the current minute is over (i.e. the first ticket belonging to the next minute has arrived)*/
623  if(minute_now > m_current_minute) // the packets with timestamp from xxxx00 to xxxx59 will belong to the same "current minute"
624  {
625  m_current_minute = minute_now;
626  uint32_t dummy_IP = 0;
627  //create a dummy ticket and send it out signaling this event
628  std::shared_ptr<TicketMsg> dummy_ticket = std::make_shared<TicketMsg>(minute_now*60, dummy_IP);
629 
630  //PER ORA STAMPARE ANCHE VIDEO IL TICKET CREATO DAL PACCHETTO CORRENTE
631  std::stringstream sss;
632  sss << "Signaling Ticket.\n"
633  << dummy_ticket.get()->to_string() << "\n\n";
634  blocklog(sss.str(), log_info);
635  send_out_through(dummy_ticket, m_out_gate);
636  }
637 
638  uint16_t transport = packet->l4_protocol(); //transport protocol (TCP or UDP or other)
639  uint32_t client_IP = 0; //source IP if uplink, destination IP if downlink
640  uint32_t server_IP = 0; //destination IP if uplink, source IP if downlink
641  uint16_t server_port = 0; //destination port if uplink, source port if downlink
642  string directionStr;
643  string modeStr;
644  string featIDStr;
645 
646  if(m_direction == UPLINK)
647  {
648  client_IP = packet->ip_src();
649  server_IP = packet->ip_dst();
650  server_port = packet->dst_port();
651  directionStr = "UL";
652 
653  }else if(m_direction == DOWNLINK)
654  {
655  client_IP = packet->ip_dst();
656  server_IP = packet->ip_src();
657  server_port = packet->src_port();
658  directionStr = "DL";
659  }
660 
661  /*TODO EXTERNAL TO THIS BLOCK:
662  * Inserire nel filtro applicato alla sorgente (blocco Pcap_source)
663  * un controllo sulla appartenenza dell'IP address:
664  * IN UPLINK (dir = src x.y.z.w)
665  * Considerare come pacchetti validi, solo i pacchetti
666  * che hanno source IP address appartenente al range di
667  * indirizzi assegnati all'operatore.
668  * IN DOWNLINK (dir = dst x.y.z.w)
669  * Considerare come pacchetti validi, solo i pacchetti
670  * che hanno destination IP address appartenente al range di
671  * indirizzi assegnati all'operatore.
672  * In questo modo sará possibile popolare le features in uplink
673  * o downlink!
674  */
675 
676  std::shared_ptr<TicketMsg> ticket = std::make_shared<TicketMsg>(packet->timestamp_s(), client_IP);
677 
678  for(int i = 0 ; i<m_num_of_ticket_features; i++)
679  {
680  //Extracting only features of the correct transport protocol
681  if(m_feat_L4_prot[i] != transport) continue;
682 
683  //Extracting features only on the user specified server port
684  if( (m_feat_ports[i] != 65535) && (m_feat_ports[i] == server_port) )
685  {
686  /************* PACKETS *************/
687  if(m_feat_modes[i] & PACKETS_ACTIVE)
688  {
689  modeStr = "packets";
690  //only if the flags match has been found, the packet contains the required feature
691  if(generate_feature_ID(packet, directionStr, m_feat_L4_prot[i], m_feat_flags[i], modeStr, m_feat_ports[i],featIDStr))
692  //key = featIDStr, value = 1
693  ticket.get()->add_feature(featIDStr, 1);
694  }
695  /************* BYTES ***************/
696  if(m_feat_modes[i] & BYTES_ACTIVE)
697  {
698  modeStr = "bytes";
699  if(generate_feature_ID(packet, directionStr, m_feat_L4_prot[i], m_feat_flags[i], modeStr, m_feat_ports[i],featIDStr))
700  //key = featIDStr, value = total length of the packet captured from the wire
701  ticket.get()->add_feature(featIDStr, packet->length());
702  }
703  /************* SERVER PORTS ***************/
704  /************ ONLY IN UPLINK *************/
705  if( (m_direction == UPLINK) && (m_feat_modes[i] & NUMPORTS_ACTIVE) )
706  {
707  modeStr = "srvPorts";
708  if(generate_feature_ID(packet, directionStr, m_feat_L4_prot[i], m_feat_flags[i], modeStr, m_feat_ports[i],featIDStr))
709  //key = featIDStr, value = the server port contacted
710  //this will be used to count the number of different server ports contacted in a timebin in the next blocks!
711  ticket.get()->add_feature(featIDStr, server_port);
712  }
713  /************* SERVER IPs ***************/
714  /************ ONLY IN UPLINK *************/
715  if( (m_direction == UPLINK) && (m_feat_modes[i] & NUMIPS_ACTIVE) )
716  {
717  modeStr = "srvIPs";
718  if(generate_feature_ID(packet, directionStr, m_feat_L4_prot[i], m_feat_flags[i], modeStr, m_feat_ports[i],featIDStr))
719  //key = featIDStr, value = the server IP address (uint32_t representation) contacted
720  //this will be used to count the number of different server IP contacted in a timebin in the next blocks!
721  ticket.get()->add_feature(featIDStr, server_IP);
722  }
723 
724  }//Extracting on all the ports but port 80
725  else if( (m_feat_ports[i] == 65535) && (server_port != 80) )
726  {
727  /************* PACKETS *************/
728  if(m_feat_modes[i] & PACKETS_ACTIVE)
729  {
730  modeStr = "packets";
731  if(generate_feature_ID(packet, directionStr, m_feat_L4_prot[i], m_feat_flags[i], modeStr, m_feat_ports[i],featIDStr))
732  ticket.get()->add_feature(featIDStr, 1);
733  }
734  /************* BYTES ***************/
735  if(m_feat_modes[i] & BYTES_ACTIVE)
736  {
737  modeStr = "bytes";
738  if(generate_feature_ID(packet, directionStr, m_feat_L4_prot[i], m_feat_flags[i], modeStr, m_feat_ports[i],featIDStr))
739  ticket.get()->add_feature(featIDStr, packet->length());
740  }
741  /************* SERVER PORTS ***************/
742  /************ ONLY IN UPLINK *************/
743  if( (m_direction == UPLINK) && (m_feat_modes[i] & NUMPORTS_ACTIVE) )
744  {
745  modeStr = "srvPorts";
746  if(generate_feature_ID(packet, directionStr, m_feat_L4_prot[i], m_feat_flags[i], modeStr, m_feat_ports[i],featIDStr))
747  ticket.get()->add_feature(featIDStr, server_port);
748  }
749  /************* SERVER IPs ***************/
750  /************ ONLY IN UPLINK *************/
751  if( (m_direction == UPLINK) && (m_feat_modes[i] & NUMIPS_ACTIVE) )
752  {
753  modeStr = "srvIPs";
754  if(generate_feature_ID(packet, directionStr, m_feat_L4_prot[i], m_feat_flags[i], modeStr, m_feat_ports[i],featIDStr))
755  ticket.get()->add_feature(featIDStr, server_IP);
756  }
757 
758  }
759 
760  }
761 
762 
763  //If at least one extraction rule matched with the current packet, output it
764  if(ticket.get()->get_extracted_features() > 0)
765  {
766  send_out_through(ticket, m_out_gate);
767 
768  //Writes the results to the output results file
769  //m_out_file << ticket.get()->to_string_file();
770 
771  //PER ORA STAMPARE ANCHE VIDEO IL TICKET CREATO DAL PACCHETTO CORRENTE
