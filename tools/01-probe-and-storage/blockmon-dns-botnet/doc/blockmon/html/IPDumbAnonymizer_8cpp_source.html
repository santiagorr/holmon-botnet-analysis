107  {
108  pugi::xml_node source=n.child("anon");
109  if(!source)
110  throw std::runtime_error("Anonymizer: missing parameter anon");
111  std::string src=source.attribute("src").value();
112  std::string dst=source.attribute("dst").value();
113  if((src.length()==0)||(dst.length()==0))
114  throw std::runtime_error("IPDumbAnonymizer: missing attribute");
115  if(src.compare("True")==0)
116  anon_src=true;
