458  void create_CCDF(vector<uint32_t> histog)
459  {
460  uint16_t Tot_num_users = 0;
461  double n_dummy_users = 0.0;
462  uint16_t n_empty_bins = 0;
463  uint i;
464  /* Counting the total number of users, including duplicates */
465  for(i = 0; i<histog.size(); i++)
466  Tot_num_users += histog[i];
467 
468  /* First Normalization */
469  for(i=0; i<histog.size(); i++)
470  {
471  m_curr_PDF.push_back((double)histog[i]/(double)Tot_num_users);
472 
473  if(m_curr_PDF[i] == 0)
474  {
475  /* Approximation */
476  m_curr_PDF[i] = EPSILON;
477  /* Significance meter: number of null bins in the histogram*/
478  n_empty_bins ++;
479  }
480  n_dummy_users += m_curr_PDF[i];
481  }
482 
483  /* Second Normalization: PDF(curr_real_histog)-> CDF-> CCDF */
484  vector<double> CDF;
485  for(i=0; i<histog.size(); i++)
486  {
487  m_curr_PDF[i] = m_curr_PDF[i]/n_dummy_users;
488 
489  if(i == 0) /*first bin*/
490  CDF.push_back(m_curr_PDF[i]);
491  else
492  CDF.push_back(CDF[i-1] + m_curr_PDF[i]);
493 
494  m_curr_CCDF.push_back(1 - CDF[i]);
495  }
496 
497  m_not_null_bins = histog.size() - n_empty_bins;
498  }

571  /*Kullback-Leibler Divergence D(p||q) function*/
572  double KuLe_Divergence(vector<double>p, vector<double>q)
573  {
574  double D_pq = 0;
575  uint i;
576  for(i=0; i < p.size(); i++)
577  D_pq += p[i]*log2(p[i]/q[i]);
578 
579  return D_pq;
580  }

583  double Entropy(vector<double> histo)
584  {
585  double Hp = 0;
586  uint i;

588  Hp += histo[i]*log2(histo[i]);

593  /* Evaluates the L_Divergence given the PDFs (p and q)*/
594  double L_Divergence(vector<double> p, vector<double>q)
595  {
596  double L_pq, D_pq, D_qp, H_p, H_q;
597  D_pq = KuLe_Divergence(p,q);
598  D_qp = KuLe_Divergence(q,p);
599  H_p = Entropy(p);
600  H_q = Entropy(q);
601  L_pq = (D_pq/H_p + D_qp/H_q)/2;
602  return L_pq;
603  }

664  /* This function evaluates the p-th percentile of the values in the array
665  * passed in input using LINEAR INTERPOLATION between the two nearest ranks. */
666  double percentile_of_array(vector<double> v_vector, uint p)
667  {
668  uint i,k = 0;
669  double v;
670  sort(v_vector.begin(), v_vector.end());
671  uint N = v_vector.size();
672  vector<double> p_vector;
673  double temp = 100/(double)N;
674 
675  for(i=0; i<N; i++)
676  p_vector.push_back(temp*(i+1-0.5));
677 
678  if(p < p_vector[0])
679  {
680  v = v_vector[0];
681  return v;
682  }
683  else if(p > p_vector[N-1])
684  {
685  v = v_vector[N-1];
686  return v;
687  }
688  else
689  {
690  for(i=0;i<N;i++)
691  if(p >= p_vector[i]) k = i;
692 
693  double n = (double)N/100;
694  v = ( v_vector[k]+n*(double)(p-p_vector[k])*(v_vector[k+1]-v_vector[k]) );
695  return v;
696  }

716  window_array RSIA_stepI(windows_hash_table::iterator w_it, const HistoMsg* histo_msg)
717  {
718  /* Guard period lower bound is setted here, the I step of RSIA
719  * starts from this bound (included) and goes until the oldest
720  * timestamp in the window */
721  uint64_t guard_period_low_bound = m_current_agg_time - m_guard_period_duration;
722 
723  window_array windows_entry = (w_it->second);
724  window_element w_el, w_elem;
725  window_array RS_Istep;
726  vector<uint> indexList;
727 
728  double dif = 0.0;
729  double sf = 0.000000000000000000; /* Slack Factor (goes from 0.0 to 0.5) */
730  uint i;
731  while ((sf <= 0.5) && (indexList.size() <= 200))
732  {
733  sf += 0.02;
734  indexList.clear();
735  for(i=0; i< windows_entry.size(); i++)
736  {
737  w_el = windows_entry[i];
738  //until the last time interval included in the window and not
739  //in the guard period
740  if(w_el.timestamp <= guard_period_low_bound)
741  {
742  dif = abs( (double)histo_msg->get_unique_users() - (double)w_el.nr_unique_users);
743 
744  if(dif < sf*(double)histo_msg->get_unique_users())
745  {
746  indexList.push_back(i);
747 
748  }
749  }
750  }
751  }
752 
753  for(i=0; i<indexList.size(); i++)
754  {
755  w_elem = windows_entry[indexList[i]];
756  RS_Istep.push_back(w_elem);
757  }
758  return RS_Istep;
759  }

773  window_array RSIA_stepII(window_array RefSet_Istep)
774  {
775  vector<Divergence> L_pq;
776  window_array RefSet_IIStep;
777  uint i;
778  for(i = 0; i < RefSet_Istep.size(); i ++)
779  {
780  window_element w_el = RefSet_Istep[i];
781  Divergence Div_el;
782  Div_el.L_div = L_Divergence(m_curr_PDF, w_el.PDF);
783  Div_el.index = i;
784  L_pq.push_back(Div_el);
785  }
786 
787  // Sorts the L_pq vector comparing the L_Divergences
788  sort(L_pq.begin(), L_pq.end(), Div_compare());
789 
790  //Checks if the array obtained from the I step has less than 50 elements
791  const uint RSI_size = RefSet_Istep.size();
792  const uint min_allowed_el = 50;
793  uint nr_of_elem = min(RSI_size, min_allowed_el);
794 
795  /* Only the first nr_of_elem (50 o less) elements with the smallest
796  * L_Divergences pass to the RSIA III step */
797  vector<Divergence> L_pq_sorted(L_pq.begin(), L_pq.begin()+nr_of_elem);
798 
799  /* Sorts the L_pq_sorted vector comparing the indexes */
800  sort(L_pq_sorted.begin(), L_pq_sorted.end(), Index_compare());
801 
802  /* Fill in the result vector */
803  for(i=0; i< L_pq_sorted.size(); i++)
804  {
805  Divergence Div_elem = L_pq_sorted[i];
806  RefSet_IIStep.push_back(RefSet_Istep[Div_elem.index]);
807  }
808 
809  return RefSet_IIStep;
810  }
