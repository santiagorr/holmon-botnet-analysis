184  template <typename H, typename ...Ts>
185  typename std::add_pointer<H>::type
186  get_header(blockmon::slice<Ts...> &s, std::true_type)
187  {
188  auto h = blockmon::get<H>(s);
189  if (h != undefinedptr)
190  return blockmon::get<H>(s);
191 
192  auto par = get_parent<H>(s);
193 
194  if (!par.first) {
195  blockmon::get<H>(s) = NULL;
196  }
197  else {
198  blockmon::get<H>(s) = reinterpret_cast<H *>(
199  static_cast<
200  typename std::conditional< std::is_const<H>::value, const char *, char *>::type
201  >(par.first) + par.second);
202  }
203 
204  return blockmon::get<H>(s);
205  }
