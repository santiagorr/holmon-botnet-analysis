34 SetReceiver* Collector::receiverForTemplate(const WireTemplate* wt) {
35  WireTemplateKey wtk = wt->key();
36 
37  // std::cerr << "looking for receiver for template:" << std::endl;
38  // wt->dump(std::cerr);
39 
40  // first look in the receiver cache
41  ReceiverCacheIter rciter = receiver_cache_.find(wtk);
42  if (rciter != receiver_cache_.end()) {
43  // std::cerr << " found in cache." << std::endl;
44  return rciter->second;
45  }
46 
47  // nope, look in the receiver list
48  for (ReceiverListIter rliter = receivers_.begin();
49  rliter != receivers_.end();
50  rliter++) {
51  if (wt->containsAll(rliter->first)) {
52  receiver_cache_[wtk] = rliter->second;
53  // std::cerr << " found in reciever list." << std::endl;
54  return rliter->second;
55  }
56  }
57 
58  // If we're here, we missed completely
59  // std::cerr << " not found" << std::endl;
60  return NULL;
61 }

63 bool Collector::receiveMessage(MBuf& mbuf) {
64  std::shared_ptr<Session> session;
65 
66  // get the next message and session from the derived class
67  if (!this->_receiveMessage(mbuf, session)) return false;
68 
69  SetReceiver* receiver;
70  Transcoder xc;
71  mbuf.transcodeBy(xc);
72 
73  // Iterate over sets
74  for (SetListIter sliter = mbuf.begin(); sliter != mbuf.end(); sliter++) {
75  const WireTemplate* set_tmpl =
76  session->getTemplate(mbuf.domain(), sliter->id);
77 
78  if (!set_tmpl->isActive()) {
79  // Inactive set template means that the template hasn't been
80  // received yet. Skip. We might want to count these later.
81  std::cerr << "No template for set " << sliter->id << "; skipping." << std::endl;
82  break;
83  }
84 
85  receiver = receiverForTemplate(set_tmpl);
86  if (receiver == NULL) {
87  // No receiver, none registered. Skip it.
88  std::cerr << "No receiver for set " << sliter->id << "; skipping." << std::endl;
89  break;
90  }
91 
92  // restrict transcoder to the set content
93  xc.focus(sliter->off + kSetHeaderLen, sliter->len - kSetHeaderLen);
94  // handle set
95  receiver->receiveSet(this, xc, set_tmpl);
96  // and defocus
97  xc.defocus();
98  }
99 
100  return true;
101 }
