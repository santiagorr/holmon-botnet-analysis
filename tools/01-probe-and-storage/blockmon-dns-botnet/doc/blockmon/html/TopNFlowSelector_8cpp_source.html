160  virtual void _receive_msg(std::shared_ptr<const Msg>&& m, int /* index */)
161  {
162  if(m->type() != MSG_ID(Flow))
163  throw std::runtime_error("TopNFlowSelector:: wrong input message type");
164  std::unique_lock<std::mutex> guard(m_mutex,std::defer_lock);
165  if(m_mutex_indirect)
166  guard.lock();
167  if(m_heap.size()<m_len)
168  {
169  m_heap.push(std::move(m));
170  return;
171  }
172  if(npackets(m)>npackets(m_heap.top()))
173  {
174  m_heap.pop();
175  m_heap.push(std::move(m));
176  }
177  }

179  /*
180  * upun timer expiration, flushes the heaviest flow messages it received during this period
181  * */
182  void _handle_timer(std::shared_ptr<Timer>&& )
183  {
184  std::unique_lock<std::mutex> guard(m_mutex,std::defer_lock);
185  if(m_mutex_indirect)
186  guard.lock();
187  while(m_heap.size()>0)
188  {
189  send_out_through(std::shared_ptr<const Msg>(std::move(m_heap.top())),m_outgate);
190  m_heap.pop();
191  }
192  }
