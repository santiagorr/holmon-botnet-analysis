220  void insert_element_in_hash_table(const TicketMsg* ticket)
221  {
222  uint32_t lookup_user = ticket->get_userID();
223  hash_table::iterator ht_it = m_tickets_ht.find(lookup_user);
224 
225  /* If the user (hash_table entry) is not yet in the hash table,
226  * the find() function returns an iterator (ht_it) pointing to the end() of
227  * the hash_table (unordered_map).
228  * Otherwise it returns the the iterator (ht_it) pointing to the found
229  * hash_table element
230  */
231  if( ht_it == m_tickets_ht.end())
232  {
233  hash_table::key_type key = lookup_user;
234  hash_table::mapped_type mapped = *(ticket->get_features());
235  hash_table::value_type val = hash_table::value_type(key, mapped);
236  m_tickets_ht.insert(val);
237 
238  }else //The user (hash_table entry) has been found
239  {
240  /* For each feature (features_map element) of the input user's ticket,
241  * we check if it already exists in the found hash_table entry.
242  */
243  for (features_map::const_iterator tck_map_it = ticket->get_features()->begin();
244  tck_map_it != ticket->get_features()->end(); ++tck_map_it)
245  {
246  string lookup_feat_ID = tck_map_it->first;
247  uint64_t lookup_feat_val = tck_map_it->second;
248 
249  features_map::iterator ht_map_it = ht_it->second.find(lookup_feat_ID);
250 
251  /* If the current feature (features_map element) has not been found.
252  * We have to insert this new element into the map
253  */
254  if(ht_map_it == ht_it->second.end())
255  {
256  ht_it->second.insert(pair<string,uint64_t>(lookup_feat_ID, lookup_feat_val));
257  }
258  else //Otherwise, the current feature already exists. We just update its value.
259  {
260  ht_map_it->second += lookup_feat_val;
261  /*********************ATTENZIONE************************
262  * SOLO NEL CASO SI STIA AGGREGANDO DALLA TIMESCALE 0
263  * IN QUALSIASI ALTRA TIMESCALE (m_input_timescale ==0)
264  * E
265  * SOLO NEL CASO IN CUI LA FEATURE É IN DIREZIONE UPLINK!!!
266  * E NECESSARIO INSERIRE QUI UNA FUNZIONE CHE CONTI
267  * OPPORTUNAMENTE IL NUMERO DI PORTE O IP ADDRESS NEL
268  * CASO LA FEATURE RIGUARDI PORTE O IP ADDRESS!
269  * INFATTI NEL CASO DI FEATURE DI QUESTO TIPO, IL VALORE
270  * DELLA COPPIA FeatID->Val CONTIENE IL NUMERO DI PORTA O
271  * L'INDIRIZZO IP. A NOI SERVE PERCIÓ CONTARE QUANTE
272  * PORTE DIVERSE O INDIRIZZI IP DIVERSI VENGONO CONTATTATI
273  * É NECESSARIO QUINDI PER CIASCUN UTENTE PREVEDERE ANCHE
274  * DELLE TABELLE CONTENENTI LE PORTE E GLI IP ADD DIVERSI
275  * CONTATTATI PER OGNI FEATURE
276  * N.B. NEL TOOL IMPLEMENTATO NEL QOE FW QUESTO NON ERA
277  * FATTO CORRETTAMENTE!!!! INFATTI IO INCREMENTAVO IL NUMERO
278  * DI PORTE E IP ADD CHE ERANO AGGREGATI SULLA SCALA DI
279  * 1 MINUTO DA PETER!
280  */
281  }

293  virtual void _receive_msg(std::shared_ptr<const Msg>&& m, int /* index */)
294  {
295  if(m->type() != MSG_ID(TicketMsg))
296  throw std::runtime_error("PerUserCounter: wrong message type");
297  const TicketMsg* ticket = static_cast<const TicketMsg* > (m.get());
298 
299  uint64_t time_now = ticket->get_timestamp();
300 
301  /* If the current time interval is not over, but an intermediate
302  * signaling (DUMMY) ticket has arrived, we need to drop the current
303  * useless ticket!
304  * This will happen if the output timescale is greater than 1 minute
305  * (from 1->5 to 30->60)
306  */
307  if( (time_now < m_current_agg_time) &&
308  ((ticket->get_userID() == 0) && (ticket->get_extracted_features() == 0))
309  ) return;
310 
311 
312  /* Since the way the dummy signaling tickets are generated in
313  * the feature extractor block, is different from the way they are
314  * generated in the current block, we need to foresee a different
315  * approach when the tickets come from the featureExtractor block
316  * (i.e. the input timescale is 0)
317  */
318  if(m_in_timescale == 0)
319  {
320 
321  /* If the current time interval is over (i.e. the first ticket (DUMMY)
322  * belonging to the next time interval has arrived)
323  * we can empty the hash table, properly update the current time
324  * (depending on the ticket just arrived and on the output timescale)
325  * forward the signaling (DUMMY) ticket setting in it the just
326  * evaluated current time.
327  *
328  * 2 checks are needed to recognize the beginning of the new time interval:
329  */
330  if( (time_now >= m_current_agg_time) && // 1. Input timestamp has to be Greater OR EQUAL to the current
331  ((ticket->get_userID() == 0) && (ticket->get_extracted_features() == 0)) // 2. The ticket is a (dummy) signaling ticket
332  )
333  {
334  /* For each element in hash table create and send a TicketMsg
335  * with timestamp m_current_time, then removes the element from the hash table
336  */
337  empty_hash_table(m_current_agg_time);
338 
339  /* Update the current time interval
340  * IMPORTANT!!! Since we are performing the division between two integers
341  * it is actually the floor of the division result!!!
342  */
343  m_current_agg_time = (time_now/(60*m_out_timescale) + 1) * (60*m_out_timescale);
344 
345  uint32_t dummy_IP = 0;
346  //create a dummy ticket and send it out signaling this event
347  std::shared_ptr<TicketMsg> dummy_ticket = std::make_shared<TicketMsg>(m_current_agg_time, dummy_IP);
348 
349  //PER ORA STAMPARE ANCHE VIDEO IL TICKET CREATO DAL PACCHETTO CORRENTE
350  std::stringstream sss;
351  sss << "Signaling Ticket. Timescale: " << m_out_timescale << " minute(s) \n"
352  << dummy_ticket.get()->to_string() << "\n\n";
353  blocklog(sss.str(), log_info);
354 
355  send_out_through(dummy_ticket, m_out_gate);
356 
357  return; //no element to insert because the current ticket is dummy!
358  }
359  }else // input timescale is not 0
360  {
361  if( (time_now > m_current_agg_time) && // 1. Input timestamp has to be Greater than the current
362  ((ticket->get_userID() == 0) && (ticket->get_extracted_features() == 0)) // 2. The ticket is a (dummy) signaling ticket
363  )
364  {
365  //for each element in hash table create and send a TicketMsg with timestamp m_current_time
366  empty_hash_table(m_current_agg_time);
367 
368  /* Update the current time interval
369  * IMPORTANT!!! Since we are performing the division between two integers
370  * it is actually the floor of the division result!!!
371  */
372  m_current_agg_time = (time_now/(60*m_out_timescale) + 1) * (60*m_out_timescale);
373 
374  uint32_t dummy_IP = 0;
375  //create a dummy ticket and send it out signaling this event
376  std::shared_ptr<TicketMsg> dummy_ticket = std::make_shared<TicketMsg>(m_current_agg_time, dummy_IP);
377 
378  //PER ORA STAMPARE ANCHE VIDEO IL TICKET CREATO DAL PACCHETTO CORRENTE
379  std::stringstream sss;
380  sss << "Signaling Ticket. Timescale: " << m_out_timescale << " minute(s) \n"
381  << dummy_ticket.get()->to_string() << "\n\n";
382  blocklog(sss.str(), log_info);
383 
384  send_out_through(dummy_ticket, m_out_gate);
385 
386  return; //no element to insert because the current ticket is dummy!
387  }
388  }
