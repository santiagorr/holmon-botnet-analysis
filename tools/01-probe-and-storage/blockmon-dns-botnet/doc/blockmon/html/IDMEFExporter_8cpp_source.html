61  void IDMEFExporter::_configure(const pugi::xml_node& n)
62  {
63  // Closes the export method if any
64  close_export_method();
65  // XML configuration
66  pugi::xml_node export_node = n.child("export");
67  pugi::xml_node file_node = n.child("file");
68  if (export_node)
69  {
70  m_export_method = EXP_TCP;
71  if (!export_node.attribute("host") || !export_node.attribute("port"))
72  throw std::runtime_error("Export specification incomplete");
73  std::string host = export_node.attribute("host").value();
74  m_host = inet_addr(host.c_str());
75  m_port = std::atoi(export_node.attribute("port").value());
76  }
77  else if (file_node)
78  {
79  m_export_method = EXP_FILE;
80  if (!file_node.attribute("name"))
81  throw std::runtime_error("File specification missing name");
82  m_filename = file_node.attribute("name").value();
83  }
84  else
85  {
86  throw std::runtime_error("No output specified");
87  }
88  // Opens the export method
89  open_export_method();
90  }

92  void IDMEFExporter::_receive_msg(std::shared_ptr<const Msg>&& m, int /* index */)
93  {
94  if(m->type() != MSG_ID(Alert)) return;
95  const Alert* alert = std::static_pointer_cast<const Alert>(m).get();
96  // Use pugixml to write the IDMEF message
97  pugi::xml_document doc;
98  pugi::xml_node decl = doc.prepend_child(pugi::node_declaration);
99  decl.append_attribute("version").set_value("1.0");
100  decl.append_attribute("encoding").set_value("utf-8");
101  pugi::xml_node message = doc.append_child("idmef:IDMEF-Message");
102  message.append_attribute("xmlns:idmef") = "http://iana.org/idmef";
103  message.append_attribute("version") = "1.0";
104  pugi::xml_node idmef_alert = message.append_child("idmef:Alert");
105  idmef_alert.append_attribute("messageid") = alert->get_identifier();
106  // Analyzer
107  pugi::xml_node analyzer = idmef_alert.append_child("idmef:Analyzer");
108  analyzer.append_attribute("analyzerid") = alert->get_analyzer().c_str();
109  // Classification
110  pugi::xml_node classification = idmef_alert.append_child("idmef:Classification");
111  classification.append_attribute("text") = alert->get_alert_name().c_str();
112  // Create time
113  time_t time = alert->get_create_time();
114  struct tm* utc_time = gmtime(&time);
115  char time_str[100];
116  strftime(time_str, 100, "%Y-%m-%dT%H:%M:%SZ", utc_time);
117  pugi::xml_node create_time = idmef_alert.append_child("idmef:CreateTime");
118  create_time.append_child(pugi::node_pcdata).set_value(time_str);
119  struct tm ntp_orig;
120  ntp_orig.tm_year = 1900;
121  ntp_orig.tm_mon = 1;
122  ntp_orig.tm_yday = 1;
123  ntp_orig.tm_hour = 0;
124  ntp_orig.tm_min = 0;
125  ntp_orig.tm_sec = 0;
126  unsigned int seconds = difftime(time, mktime(&ntp_orig));
127  char ntp_timestamp[22];
128  sprintf(ntp_timestamp, "0x%x.0x0", seconds);
129  create_time.append_attribute("ntpstamp") = ntp_timestamp;
130  // Assessment
131  if (alert->is_assessment_set()) {
132  pugi::xml_node assessment = idmef_alert.append_child("idmef:Assessment");
133  pugi::xml_node impact = assessment.append_child("idmef:Impact");
134  const char* severity;
135  switch (alert->get_severity()) {
136  case Alert::sev_high:
137  severity = "high";
138  break;
139  case Alert::sev_medium:
140  severity = "medium";
141  break;
142  case Alert::sev_low:
143  severity = "low";
144  break;
145  case Alert::sev_info:
146  severity = "info";
147  break;
148  default:
149  severity = "medium";
150  break;
151  }
152  impact.append_attribute("severity") = severity;
153  pugi::xml_node confidence = assessment.append_child("idmef:Confidence");
154  const char* conf_str;
155  switch (alert->get_confidence()) {
156  case Alert::conf_high:
157  conf_str = "high";
158  break;
159  case Alert::conf_medium:
160  conf_str = "medium";
161  break;
162  case Alert::conf_low:
163  conf_str = "low";
164  break;
165  default:
166  conf_str = "medium";
167  break;
168  }
169  confidence.append_attribute("confidence") = conf_str;
170  }
171  // Sources
172  const std::vector<Alert::Node>* sources = alert->get_sources();
173  for (unsigned int i = 0; i < sources->size(); i++) {
174  pugi::xml_node node = idmef_alert.append_child("idmef:Source");
175  set_xml_from_node(node, sources->at(i));
176  }
177  // Targets
178  const std::vector<Alert::Node>* targets = alert->get_targets();
179  for (unsigned int i = 0; i < targets->size(); i++) {
180  pugi::xml_node node = idmef_alert.append_child("idmef:Target");
181  set_xml_from_node(node, targets->at(i));
182  }
183  // Send the IDMEF message to a file or using TCP
184  if (m_export_method == EXP_FILE) {
185  // Export to a file
186  doc.save(m_file_stream);
187  m_file_stream.flush();
188  } else if (m_export_method == EXP_TCP) {
189  // Export through a TCP socket
190  std::ostringstream stream;
191  doc.save(stream);
192  stream.flush();
193  std::string str_doc = stream.str();
194  if (send(m_tcp_socket, str_doc.c_str(), str_doc.size() + 1, 0) < 0)
195  throw std::runtime_error("Impossible to send an alert through the TCP connection");
196  }
197  }

199  void IDMEFExporter::set_xml_from_node(pugi::xml_node& xml_node, const Alert::Node& node) {
200  pugi::xml_node idmef_node = xml_node.append_child("idmef:Node");
201  if (node.ipv4_set()) {
202  pugi::xml_node address = idmef_node.append_child("idmef:Address");
203  address.append_attribute("category") = "ipv4-addr-hex";
204  pugi::xml_node ip = address.append_child("idmef:address");
205  char ipv4[11];
206  sprintf(ipv4, "0x%x", node.get_ipv4());
207  ip.append_child(pugi::node_pcdata).set_value(ipv4);
208  }
209  if (node.domain_name_set()) {
210  idmef_node.append_attribute("category") = "dns";
211  pugi::xml_node dns_name = idmef_node.append_child("idmef:name");
212  dns_name.append_child(pugi::node_pcdata).set_value(node.get_domain_name().c_str());
213  }

216  void IDMEFExporter::open_export_method() {
217  if (m_export_method == EXP_FILE) {
218  // Open the file for writing
219  m_file_stream.open(m_filename, std::ios::out | std::ios::app);
220  m_file_open = true;
221  } else if (m_export_method == EXP_TCP) {
222  // Open the TCP socket
223  m_tcp_socket = socket(AF_INET, SOCK_STREAM, 0);
224  if(m_tcp_socket == -1)
225  throw std::runtime_error("Impossible to open the TCP connection");
226  sockaddr_in sock_addr = { 0 };
227  sock_addr.sin_addr.s_addr = m_host;
228  sock_addr.sin_port = htons(m_port);
229  sock_addr.sin_family = AF_INET;
230  if(connect(m_tcp_socket, (sockaddr *) &sock_addr, sizeof(sockaddr)) == -1)
231  throw std::runtime_error("No server is accepting the TCP connection");
232  m_tcp_open = true;
233  }
234  }
