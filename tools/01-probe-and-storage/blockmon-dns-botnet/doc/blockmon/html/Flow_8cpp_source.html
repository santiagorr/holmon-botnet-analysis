36  size_t Flow::get_payload(uint8_t* buffer, size_t offset, size_t nbytes) {
37  if (m_flow_packets.empty())
38  return 0;

40  /* Doing TCP stream reassembly by sorting the list on the
41  * sequence number does not work if those sequence numbers wrap
42  * around. It also does not work in the presence of adversaries
43  * who do funky stuff in order to make TCP reassembly difficult
44  * on purpose.
45  *
46  * The problem is that we have too little information about the
47  * flow. For example, we don't know the receiver's window size,
48  * only the sender's. So what we do is this:
49  *
50  * 1. Go through the list and find the (one) packet with a SYN.
51  * Let its sequence number be s_syn.
52  * 2. Build a second list (because we can't modify the original
53  * packets) consisting of pairs (i, s), where s is the
54  * sequence number of packet i, minus s_syn.
55  * 3. Sort that second list on the modified sequence number.
56  *
57  * If it is possible that the original packets do not contain a
58  * SYN, the following algorithm will also work, but isn't as
59  * neat:
60  *
61  * 1. Sort the list by TCP sequence number.
62  * 2. Go through the list. If the list has exactly one "hole"
63  * in it, and one that would be expected by a TCP sequence
64  * number wraparound, we swap the twp list pieces that are
65  * "around" that hole.
66  *
67  * For example, let's say the TCP sequence numbers and packet
68  * lengths after sorting are (50, 50), (100, 100), (200, 100),
69  * (2^32-200, 100), (2^32-100, 150). Then there's a big hole
70  * between 200 and 2^32-200, and one that we would expect from a
71  * sequence number wraparound, from which we conclude that the
72  * correct sequence of packets must have the sequence numbers
73  * 2^32-200, 2^32-100, 0, 100, 200, in that order.
74  *
75  * At the same time, we could build a list of offsets that will
76  * be nice to have later on. Consider the above example again.
77  * In this case, the offset list would be o = <0, 150, 200, 300,
78  * 400>, and o[i] would give the offset in the TCP payload where
79  * this packet's payload starts.
80  */
81  enum { packets_tcp, packets_udp, packets_unknown } packet_type = packets_unknown;
82  auto syn = m_flow_packets.end();
83  if (!m_list_sorted) {
84  bool first = true;
85 
86  for (auto i = m_flow_packets.begin();
87  i != m_flow_packets.end();
88  ++i) {
89  const Packet* packet = i->get();
90 
91  if (!packet->is_tcp() && !packet->is_udp())
92  throw std::runtime_error("FlowMeter: can get flow payload "
93  "only for TCP or UDP flows at "
94  "the moment");;
95  if (packet_type == packets_unknown)
96  packet_type = (packet->is_tcp() ? packets_tcp : packets_udp);
97  else if ((packet_type == packets_udp && packet->is_tcp() )
98  || (packet_type == packets_tcp && packet->is_udp()))
99  throw std::runtime_error("FlowMeter: flow contains both TCP"
100  " and UDP packets");
101  else if (packet_type == packets_udp && !first)
102  throw std::runtime_error("FlowMeter: flow contains more than"
103  " one UDP packet");
104  else if (packet_type == packets_tcp && packet->tcp_has_syn())
105  syn = i;
106  first = false;
107  }
108  if (packet_type == packets_tcp) {
109  m_flow_packets.sort([](const std::shared_ptr<const Packet> a,
110  const std::shared_ptr<const Packet> b) {
111  return a->tcp_seq() < b->tcp_seq();
112  });
113 
114  /* You may ask yourself, is the SYN variable, containing an
115  * iterator to the packet with a SYN, still valid after the
116  * call to sort()? Well, here's what the C++ standard says
117  * in N3242 in Section 23.2.1, clause 10:
118  *
119  * "Unless otherwise specified (either explicitly or by
120  * defining a function in terms of other functions),
121  * invoking a container member function or passing a
122  * container as an argument to a library function shall not
123  * invalidate iterators to, or change the values of, objects
124  * within that container."
125  *
126  * And the documentation for std::list::sort() (Section
127  * 23.3.5.5, clause 25) contains no such explicit
128  * specification or definition in terms of another
129  * function.
130  */
131  if (syn != m_flow_packets.begin()) {
132  /* Not already sorted. This would also work if the list
133  * is in fact already sorted, but it would do it with a
134  * fairly expensive no-op. */
135  std::list<std::shared_ptr<const Packet>> last_part;
136  last_part.splice(last_part.begin(), // before this point,
137  m_flow_packets, // move items from this list
138  m_flow_packets.begin(), // starting here
139  syn); // and ending just before here
140  // Now append LAST_PART to M_FLOW_PACKETS
141  m_flow_packets.splice(m_flow_packets.end(), last_part);
142  }
143  }
144  m_list_sorted = true;
145  }

148  size_t effective_offset = offset;
149  size_t bytes_skipped = 0;
150  size_t bytes_copied = 0;
151  uint8_t *p = buffer; /* The invariant is bytes_skipped +
152  bytes_copied = p - buffer */
153  size_t bytes_remaining = nbytes;
154  uint32_t expected_sequence_number = 0;
155  bool first = true;
156  bool more = true;
157 
158  for (auto i = m_flow_packets.begin();
159  i != m_flow_packets.end() && more && bytes_remaining > 0;
160  ++i) {
161  const Packet* packet = i->get();
162  size_t payload_length = 0;
163  if (first) {
164  if (packet->is_udp()) {
165  effective_offset = 0;
166  } else if (packet->is_tcp()) {
167  if (packet->tcp_has_syn()) {
168  expected_sequence_number = packet->tcp_seq();
169  } else // Flow starts with a hole: abort, abort!
170  more = false;
171  } else
172  throw std::runtime_error("FlowMeter: can get flow payload "
173  "only for TCP or UDP flows at "
174  "the moment");;
175  first = false;
176  }
177 
178  if (packet->is_tcp())
179  payload_length = packet->ip_length() - packet->ip_header_length()
180  - packet->tcp_header_length();
181  else if (packet->is_udp())
182  payload_length = packet->udp_length() - sizeof(udphdr);
183  else
184  throw std::runtime_error("FlowMeter: subsequent packet in flow "
185  "neither TCP nor UDP");;
186 
187  if (more) {
188  if (bytes_skipped + bytes_copied + payload_length > effective_offset) {
189  size_t bytes_to_copy = std::min<size_t>(bytes_remaining,
190  payload_length);
191  size_t payload_offset =
192  bytes_skipped + bytes_copied == 0 ? effective_offset : 0;
193  std::memcpy(p, packet->payload() + payload_offset, bytes_to_copy);
194  bytes_copied += bytes_to_copy;
195  bytes_remaining -= bytes_to_copy;
196  p += bytes_to_copy;
197  bytes_skipped = offset;
198  } else
199  bytes_skipped += payload_length;
200 
201  if (packet->is_tcp()) {
202  if (packet->tcp_seq() == expected_sequence_number)
203  /* Unsigned arithmetic is modulo arithmetic, so this
204  * addition will produce the correct wraparound effect. */
205  expected_sequence_number += payload_length + 1;
206  else // Hole
207  more = false;
208  } else if (packet->is_udp())
209  /* Can only be one packet. */
210  more = false;
211  else
212  throw std::runtime_error("FlowMeter: subsequent packet in flow "
213  "neither TCP nor UDP (can't happen, "
214  "but apparently did)");
215  }
216  }
217  return bytes_copied;
218  }
