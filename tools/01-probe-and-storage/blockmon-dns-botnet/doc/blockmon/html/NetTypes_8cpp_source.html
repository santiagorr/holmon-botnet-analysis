84  const std::string ipv4_to_string(uint32_t ipv4) {
85  std::ostringstream os;

87  << ((ipv4 & 0x00FF0000) >> 16) << "."

93  const uint32_t string_to_ipv4(std::string str) {
94  std::vector<std::string> strvec;
95  boost::split(strvec, str, boost::is_any_of("."));
96  uint32_t a = boost::lexical_cast<uint32_t>(strvec[0]) << 24;
97  a |= (boost::lexical_cast<uint32_t>(strvec[1]) << 16);
98  a |= (boost::lexical_cast<uint32_t>(strvec[2]) << 8);
99  a |= boost::lexical_cast<uint32_t>(strvec[3]);
