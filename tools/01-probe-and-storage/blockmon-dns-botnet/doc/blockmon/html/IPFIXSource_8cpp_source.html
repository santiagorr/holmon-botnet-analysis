133  virtual void _configure(const pugi::xml_node& params_node) {
134  bootstrapIPFIX();
135 
136  // Does not support reconfiguration yet.
137  if (m_collector) {
138  throw new std::logic_error("IPFIXSource not reconfigurable");
139  }
140 
141  // create a collector
142  pugi::xml_node col_node;
143  if ((col_node = params_node.child("collect"))) {
144  std::string transport = col_node.attribute("transport").value();
145  std::string port = col_node.attribute("port").value();
146 
147  if (transport.empty()) {
148  transport = "tcp";
149  }
150  if (port.empty()) {
151  port = std::string("4739");
152  }
153 
154  if (transport == std::string("udp")) {
155  configUDPCollection(port);
156  } else if (transport == std::string("tcp")) {
157  configTCPCollection(port);
158  } else if (transport == std::string("sctp")) {
159  throw std::runtime_error("SCTP not yet supported");
160  } else {
161  throw std::runtime_error("Unsupported transport");
162  }
163  } else if ((col_node = params_node.child("file"))) {
164  std::string name = col_node.attribute("name").value();
165  if (name.empty()) {
166  throw std::runtime_error("File requires name");
167  }
168  configFileImport(name);
169  }
170 
171  // Populate list of bridges (data types)
172  for (pugi::xml_node dt_node = params_node.child("datatype");
173  dt_node;
174  dt_node = dt_node.next_sibling("datatype"))
175  {
176  std::string dt_name = dt_node.attribute("name").value();
177  if (dt_name.empty()) {
178  throw std::runtime_error("Datatype missing name");
179  }
180 
181  std::shared_ptr<IPFIXTypeBridge> bridge =
182  IPFIXTypeRegistry::instance().bridgeForTypeName(dt_name);
183  if (bridge.get() == NULL) {
184  throw std::runtime_error("Unknown datatype " + dt_name);
185  }
186 
187  // prepare the collector (register receivers)
188  bridge->prepareCollector(*m_collector, get_output_gate(m_gate_id));
189 
190  // store the bridge
191  m_bridges.push_back(bridge);
192  }
193  }

204  void _do_async() {
205  if (m_collector) {
206  if (!m_collector->receiveMessage(m_mbuf)) {
207  // null out collector -- causes close
208  // FIXME yeah sick hackery but g++ is tedious
209  m_collector = std::unique_ptr<IPFIX::Collector>(reinterpret_cast<IPFIX::Collector*>(0));
210 
211  blocklog(std::string("Collector at end of stream, will wait ") +
212  boost::lexical_cast<std::string>(m_eosdelay) +
213  " seconds, then shut down", log_warning);
214  // set a timer to shut down after a delay
215  set_timer_at(get_BM_time() + m_eosdelay * 1000000, "end of stream", kEOSTimer);
216  }
217  } else {
218  // async without collector, do nothing.
219  }
220  }

224  blocklog(std::string("Collector shutdown"), log_warning);
225  exit(0);
226  }
