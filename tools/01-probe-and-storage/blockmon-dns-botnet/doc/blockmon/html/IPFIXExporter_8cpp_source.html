143  virtual void _configure(const pugi::xml_node& params_node) {
144  uint32_t odid = 1;
145 
146  bootstrapIPFIX();
147 
148  // Does not support reconfiguration yet.
149  if (m_exporter) {
150  throw new std::logic_error("IPFIXExporter not reconfigurable");
151  }
152 
153  // Get optional observation domain ID (default 1)
154  pugi::xml_node domain_node;
155  if ((domain_node = params_node.child("domain"))) {
156  std::string domain_id = domain_node.attribute("id").value();
157  if (domain_id.empty()) {
158  throw std::runtime_error("Domain missing ID");
159  }
160  odid = boost::lexical_cast<uint32_t>(domain_id);
161  }
162 
163  // create an exporter
164  pugi::xml_node exp_node;
165  if ((exp_node = params_node.child("export"))) {
166  // Parse export (transport, host, port)
167  std::string transport = exp_node.attribute("transport").value();
168  std::string host = exp_node.attribute("host").value();
169  std::string port = exp_node.attribute("port").value();
170 
171  if (transport.empty()) {
172  transport = "tcp";
173  }
174  if (host.empty()) {
175  throw std::runtime_error("Export requires host");
176  }
177  if (port.empty()) {
178  port = std::string("4739");
179  }
180 
181  if (transport == std::string("udp")) {
182  configUDPExport(odid, host, port);
183  } else if (transport == std::string("tcp")) {
184  configTCPExport(odid, host, port);
185  } else if (transport == std::string("sctp")) {
186  throw std::runtime_error("SCTP not yet supported");
187  } else {
188  throw std::runtime_error("Unsupported transport");
189  }
190  } else if ((exp_node = params_node.child("file"))) {
191  std::string name = exp_node.attribute("name").value();
192  if (name.empty()) {
193  throw std::runtime_error("File requires name");
194  }
195  configFileExport(odid, name);
196  }
197 
198  // Set fast flush mode if required
199  if (params_node.child("fastflush")) {
200  m_exporter->setFastFlush();
201  }
202 
203  // Populate list of bridges (data types)
204  for (pugi::xml_node dt_node = params_node.child("datatype");
205  dt_node;
206  dt_node = dt_node.next_sibling("datatype"))
207  {
208  std::string dt_name = dt_node.attribute("name").value();
209  if (dt_name.empty()) {
210  throw std::runtime_error("Datatype missing name");
211  }
212 
213  std::shared_ptr<IPFIXTypeBridge> bridge =
214  IPFIXTypeRegistry::instance().bridgeForTypeName(dt_name);
215  if (bridge.get() == NULL) {
216  throw std::runtime_error("Unknown datatype " + dt_name);
217  }
218 
219  // prepare the exporter (populate templates)
220  bridge->prepareExporter(*m_exporter);
221 
222  // store the bridge
223  m_bridges.push_back(bridge);
224  }
225 
226  // export all templates
227  m_exporter->exportTemplatesForDomain();
228  }

230  virtual void _receive_msg(std::shared_ptr<const Msg>&& m, int gate_id) {
231 
232  // select the first suitable bridge for the message
233  for (auto iter = m_bridges.begin(); iter != m_bridges.end(); iter++) {
234  if ((*iter)->canExport(m)) {
235  // yep, export and leave.
236  (*iter)->exportMsg(m, *m_exporter);
237  return;
238  }
239  }
240 
241  blocklog("no suitable datatype for message: not exported", log_warning);
242  }
