35  void WireTemplate::add(const InfoElement* ie) {
36  add_inner(ie);
37 
38  // Add offset to offset table
39  offsets_.push_back(nextoff_);
40 
41  // calculate next offset
42  if (nextoff_ != kVarlen) {
43  max_fixed_offset_ = nextoff_;
44  nextoff_ += ie->len();
45  }
46 
47  // Add the length of the IE to the minimum length
48  if (ie->len() == kVarlen) {
49  minlen_ += 1;
50  varlen_count_ += 1;
51  nextoff_ = kVarlen;
52  } else {
53  minlen_ += ie->len();
54  }
55 
56  // Add the length of the IE's fields to the template record length
57  trlen_ += ie->pen() ? 8 : 4;
58  }

77 bool WireTemplate::encodeStruct(Transcoder& xc, const StructTemplate& struct_tmpl, uint8_t* struct_cp) const
78 {
79  // Refuse to encode an with inactive template
80  assert(active_);
81 
82  // Refuse to encode unless we have at _least_ minimum length avail.
83  if (xc.avail() < minlen_) {
84  //std::cerr << "encoder overrun (" << xc.avail() << " avail, " << minlen_ << " required.)" << std::endl;
85  return false;
86  }
87 
88  // Note: Void casts here are safe, since we checked minlen.
89  xc.checkpoint();
90  for (IETemplateIter iter = ies_.begin();
91  iter != ies_.end();
92  iter++) {
93  if (struct_tmpl.contains(*iter)) {
94  size_t off = struct_tmpl.offset(*iter);
95  size_t len = struct_tmpl.length(*iter);
96  if (len == kVarlen) {
97  VarlenField *vf = reinterpret_cast<VarlenField *>(struct_cp + off);
98  if (!xc.encode(*vf, *iter)) goto err;
99  } else {
100  if (!xc.encode(struct_cp + off, len, *iter)) goto err;
101  }
102  } else {
103  if (!xc.encodeZero(*iter)) goto err;
104  }
105  }
106 
107  return true;
108 
109 err:
110  xc.rollback();
111  return false;
112 }

115 bool WireTemplate::encodeTemplateRecord(Transcoder &xc) const {
116 
117  // Refuse to encode an inactive template
118  assert(active_);
119 
120  // Make sure the encoder has space for the encoded version of the template
121  // Void casts from here are safe, since the template record length is
122  // constant.
123  if (xc.avail() < trlen_) {
124  return false;
125  }
126 
127  // encode template record header
128  (void) xc.encode(tid_);
129  (void) xc.encode(static_cast<uint16_t>(ies_.size()));
130 
131  // encode fields
132  for (IETemplateIter iter = ies_.begin();
133  iter != ies_.end();
134  iter++) {
135  if ((*iter)->pen()) {
136  (void) xc.encode(static_cast<uint16_t>((*iter)->number() | kEnterpriseBit));
137  (void) xc.encode((*iter)->len());
138  (void) xc.encode((*iter)->pen());
139  } else {
140  (void) xc.encode((*iter)->number());
141  (void) xc.encode((*iter)->len());
142  }
143  }
144 
145  return true;
146 }
