340  void configure(const pugi::xml_node& block_node)
341  {
342  // configure needs to take <block> node of XML
343  // parsing common block features
344 #ifdef BLOCKING_QUEUE
345  auto block_attrib=block_node.attribute("blocking_mode");
346  if(!block_attrib.empty()){
347  if (m_invocation != invocation_type::Indirect){
348  std::cerr << "warning: m_queue_blocking attribute is valid only for indirect blocks. The argument is ignored for the block " <<m_name<<"\n";
349  }
350  else
351  {
352  std::string s(block_attrib.value());
353  std::transform(s.begin(), s.end(), s.begin(), (int (*)(int))std::tolower);
354  if (s.compare("drop")==0){
355  m_queue_type=QUEUETYPE_DROPPING;
356  blocklog("Queues are dropping", log_info);
357  }
358  else if (s.compare("yield")==0){
359  m_queue_type=QUEUETYPE_YIELDING;
360  blocklog("Queues are blocking (threads yield)", log_info);
361  }else if(s.compare("sleep")==0){
362  m_queue_type=block_node.attribute("sleep_usec").as_int(); // returns 0 if sleep_usec is not set
363  if(m_queue_type<=0){
364  throw std::runtime_error(std::string("For block '").append(m_name).append("': 'blocking_mode=sleep' but 'sleep_usec' attribute not set or not > 0"));
365  }
366  blocklog(std::string("Queues are blocking (threads sleep for ").append(std::to_string(m_queue_type)).append(" usec)"), log_info);
367 
368  }else if(s.compare("mutex")==0){
369  m_queue_type=QUEUETYPE_MUTEX;
370  blocklog(std::string("Queues are blocking (using signals and a mutex)"), log_info);
371  }else{
372  throw std::runtime_error(std::string("'blocking_mode' attribute value '").append(s)
373  .append("' for block '").append(m_name).append("' is invalid.\n")
374  .append("Valid parameters are: drop (default), yield, sleep, mutex.") );
375  }
376  }
377  }
378  else if (m_invocation != invocation_type::Indirect){
379  blocklog("Queues are non-blocking (i.e., messages will be dropped when queue is full)", log_info);
380  m_queue_type=QUEUETYPE_DROPPING;
381  }
382 
383 #endif //BLOCKING_QUEUE
384  pugi::xml_node params_node = block_node.child("params");
385 
386  _configure(params_node);
387  m_issynchronized = _synchronize_access();
388  }
