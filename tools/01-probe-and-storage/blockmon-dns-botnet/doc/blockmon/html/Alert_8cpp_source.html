126  std::string Alert::to_string() const
127  {
128  std::stringstream ss;
129  ss << "Alert " << m_alert_name << " from " << m_analyzer;
130  ss << " (" << m_identifier << "): ";
131  for (unsigned int i = 0; i < m_sources.size(); i++) {
132  if (i > 0)
133  ss << ", ";
134  ss << ip_to_string(m_sources[i].get_ipv4());
135  }
136  ss << " -> ";
137  for (unsigned int i = 0; i < m_targets.size(); i++) {
138  if (i > 0)
139  ss << ", ";
140  ss << ip_to_string(m_targets[i].get_ipv4());
141  }
142  return ss.str();
143  }

145  std::string Alert::ip_to_string(unsigned int ip) const
146  {
147  std::stringstream ss;

149  unsigned int tmp_val = tmp_char[3];

154  ss << "." << tmp_val;
155  tmp_val = tmp_char[0];
156  ss << "." << tmp_val;
157  return ss.str();
