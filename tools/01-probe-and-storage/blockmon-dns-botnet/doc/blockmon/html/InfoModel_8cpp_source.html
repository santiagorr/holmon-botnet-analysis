40 static void parseIESpec_NumPen(std::istringstream& iestream,
41  unsigned int& number,
42  unsigned int& pen) {
43  // get the first number
44  iestream >> number;
45  if (iestream.fail())
46  throw IESpecError("badly formatted IE number / PEN");
47 
48  // see how we're terminated
49  char c = iestream.get();
50  if (c == '/') {
51  // first number was a pen. get the next.
52  pen = number;
53  iestream >> number;
54  if (iestream.fail())
55  throw IESpecError("badly formatted IE number");
56  } else {
57  pen = 0;
58  iestream.unget();
59  }
60 }

62 static void parseIESpec_Length(std::istringstream& iestream,
63  unsigned int& len) {
64 
65  std::stringbuf lenbuf;
66 
67  iestream.get(lenbuf, ']');
68 
69  if (lenbuf.str()[0] == 'v') {
70  len = kVarlen;
71  } else { // parse size
72  unsigned long ullen;
73  try {
74  ullen = std::stoul(lenbuf.str());
75  } catch (std::invalid_argument) {
76  throw IESpecError("bad size " + lenbuf.str() + " (invalid format)");
77  } catch (std::out_of_range) {
78  throw IESpecError("bad size " + lenbuf.str() + " (out of range)");
79  }
80 
81  if (ullen != kVarlen &&
82  ullen > kVarlen - kMessageHeaderLen - kSetHeaderLen)
83  {
84  throw IESpecError("bad size " + std::to_string(ullen)
85  + " (too large)");
86  }
87 
88  len = static_cast<unsigned int>(ullen);
89  }
90 }

92 static void match(std::istringstream& iestream, char x) {
93  char c = iestream.get();
94  if (iestream.fail()) {
95  std::ostringstream b;
96  b << "expected character '" << x << "', but read failed";
97  throw IESpecError(b.str().c_str());
98  } else if (c != x) {
99  std::ostringstream b;
100  b << "expected character '" << x << "', but got '" << c << "'";
101  throw IESpecError(b.str().c_str());
102  }
103 }

105 static void parseIESpec_Initial(std::istringstream& iestream,
106  std::stringbuf& namebuf, bool& name_set,
107  std::stringbuf& typebuf, bool& type_set,
108  std::stringbuf& ctxbuf, bool& ctx_set,
109  unsigned int& number, bool& number_set,
110  unsigned int& pen, bool& pen_set,
111  unsigned int& len, bool& len_set)
112 {
113  char c = iestream.get();
114  if (iestream.eof()) return;
115 
116  switch (c) {
117  case '(':
118  assert(!pen_set || number_set);
119  if (number_set)
120  throw IESpecError("IESpec contains number / pen more than once");
121 
122  parseIESpec_NumPen(iestream, number, pen);
123  match(iestream, ')');
124  number_set = true;
125  pen_set = pen != 0;
126  break;
127  case '[':
128  if (len_set)
129  throw IESpecError("IESpec contains length more than once");
130  parseIESpec_Length(iestream, len);
131  match(iestream, ']');
132  len_set = true;
133  break;
134  case '<':
135  if (type_set)
136  throw IESpecError("IESpec contains type more than once");
137  iestream.get(typebuf, '>');
138  match(iestream, '>');
139  type_set = true;
140  break;
141  case '{':
142  if (ctx_set)
143  throw IESpecError("IESpec contains contextf more than once");
144  iestream.get(ctxbuf, '}');
145  match(iestream, '}');
146  ctx_set = true;
147  break;
148  default:
149  if (name_set)
150  throw IESpecError("IESpec contains name more than once");
151  do {
152  namebuf.sputc(c);
153  c = iestream.get();
154  } while (!iestream.eof()
155  && c != '(' && c != '{' && c != '<' && c != '[');
156  if (!iestream.eof())
157  iestream.unget();
158  name_set = true;
159  break;
160  }
161 }

163 const InfoElement InfoModel::parseIESpec(const std::string& iespec) const {
164 
165  // check for name-only IE
166  // WORKAROUND for broken libc++ on Mac OS X Lion
167  if ((iespec.find('(') == std::string::npos) &&
168  (iespec.find('[') == std::string::npos) &&
169  (iespec.find('<') == std::string::npos) &&
170  (iespec.find('{') == std::string::npos))
171  {
172  InfoElement ie(iespec, 0, 0, 0, 0);
173  return ie;
174  }
175 
176  std::istringstream iestream(iespec);
177  std::stringbuf namebuf, typebuf, ctxbuf;
178  unsigned int number = 0, pen = 0, len = 0;
179  bool name_set = false;
180  bool type_set = false;
181  bool ctx_set = false;
182  bool number_set = false;
183  bool pen_set = false;
184  bool len_set = false;
185 
186  while (!iestream.eof()) {
187  parseIESpec_Initial(iestream,
188  namebuf, name_set,
189  typebuf, type_set,
190  ctxbuf, ctx_set,
191  number, number_set,
192  pen, pen_set,
193  len, len_set);
194  }
195 
196  const IEType *ietype = lookupIEType(typebuf.str());
197  InfoElement ie(namebuf.str(), pen, number, ietype, len);
198  return ie;
199 }

277 void InfoModel::add(const InfoElement& ie) {
278  // Short circuit unless we have a record valid for insertion:
279  // at least a name, number, and valid, known type
280  if (!ie.name().size() || !ie.number() || ie.ietype() == IEType::unknown()) {
281  throw IESpecError("incomplete IESpec for InfoModel addition");
282  }
283 
284  // Only add if we don't have an existing IE for the given name and pen
285  if (lookupIE(ie.pen(), ie.number(), ie.len())) return;
286 
287  if (ie.pen()) {
288  name_registry_[ie.name()] =
289  pen_registry_[ie.pen()][ie.number()] =
290  std::shared_ptr<InfoElement>(new InfoElement(ie));
291  // std::cerr << "add PEN IE " << ie.pen() << "/" << ie.number() << " " << ie.name() << std::endl;
292  } else {
293  name_registry_[ie.name()] =
294  iana_registry_[ie.number()] =
295  std::shared_ptr<InfoElement>(new InfoElement(ie));
296  // std::cerr << "add IANA IE " << ie.number() << " " << ie.name() << std::endl;
297  }
298 }

313 const InfoElement* InfoModel::lookupIE(uint32_t pen, uint16_t number, uint16_t len) const {
314  std::map<uint16_t, std::shared_ptr<InfoElement> >::const_iterator iter;
315 
316  //std::cerr << "lookupIE (" << pen << "/" << number << ")[" << len << "]" << std::endl;
317  if (pen) {
318  std::map<uint32_t, std::map<uint16_t, std::shared_ptr<InfoElement> > >::const_iterator peniter;
319 
320  if ((peniter = pen_registry_.find(pen)) == pen_registry_.end()) {
321  //std::cerr << " no such pen" << std::endl;
322  return NULL;
323  } else {
324  if ((iter = peniter->second.find(number)) == peniter->second.end()) {
325  //std::cerr << " not in pen registry" << std::endl;
326  return NULL;
327  }
328  }
329  } else {
330  if ((iter = iana_registry_.find(number)) == iana_registry_.end()) {
331  //std::cerr << " not in iana registry" << std::endl;
332  return NULL;
333  }
334  }
335 
336  return iter->second->forLen(len);
337 }

339 const InfoElement *InfoModel::lookupIE(const InfoElement& specie) const {
340 
341  if (specie.number()) {
342  return lookupIE(specie.pen(), specie.number(), specie.len());
343  } else if (specie.name().empty()) {
344  // Nothing to look up.
345  throw IESpecError("incomplete IESpec for InfoModel lookup.");
346  } else {
347  // std::cerr << "lookupIE " << specie.name() << std::endl;
348  std::map<std::string, std::shared_ptr<InfoElement> >::const_iterator iter = name_registry_.find(specie.name());
349  if (iter == name_registry_.end()) {
350  // std::cerr << " not in name registry" << std::endl;
351  return NULL;
352  } else {
353  return iter->second->forLen(specie.len());
354  }
355  }
356 }
