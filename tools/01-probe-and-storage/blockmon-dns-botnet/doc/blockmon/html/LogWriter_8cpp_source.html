158  virtual void _receive_msg(std::shared_ptr<const Msg>&& m, int /* index */){
159  if (m->type() != m_msg_type)
160  throw std::runtime_error("LogWriter:: wrong message type");
161  lineno++;
162  endTime = time(NULL);
163  if(h_res_gng == TAG_INVALID)
164  throw std::runtime_error("LogWriter: invalid tag handle");
165 
166  const char* value;
167  int counter = 0;
168  while ( (value = (const char* ) m->get_tag<boost::array<char,SIZE_MSG> >(h_res_gng)) == NULL)
169  {
170  usleep(1);
171  if (counter == UTIME_WAIT)
172  throw std::runtime_error("LogWriter: max UTIME_WAIT");
173  counter++;
174  }
175  fout << "," << *value << ")\t";
176 
177  counter = 0;
178  while ( (value = (const char* ) m->get_tag<boost::array<char,SIZE_MSG> >(h_line)) == NULL)
179  {
180  usleep(1);

182  throw std::runtime_error("LogWriter: max UTIME_WAIT");

184  }
