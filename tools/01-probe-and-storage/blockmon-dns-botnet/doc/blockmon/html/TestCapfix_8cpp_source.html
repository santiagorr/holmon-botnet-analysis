96 void exportPacket (unsigned char *evp,
97  const pcap_pkthdr* caphdr,
98  const uint8_t* capbuf) {
99  FileWriter* e = reinterpret_cast<FileWriter *>(evp);
100  CapfixPacket pkt;
101  size_t caplen = caphdr->caplen;
102 
103  if (datalink == DLT_EN10MB) {
104  const ethhdr* eh = reinterpret_cast<const ethhdr*>(capbuf);
105  if (ntohs(eh->ethertype) == kEthertypeIP4) {
106  caplen -= sizeof(ethhdr);
107  capbuf += sizeof(ethhdr);
108  } else {
109  return;
110  }
111  }
112 
113  pkt.observationTimeMilliseconds = (uint64_t)caphdr->ts.tv_sec * 1000;
114  pkt.observationTimeMilliseconds += caphdr->ts.tv_usec / 1000;
115  pkt.ipTotalLength = caphdr->len;
116  pkt.ipHeaderPacketSection.len = (size_t)caphdr->caplen;
117  pkt.ipHeaderPacketSection.cp = (uint8_t *)capbuf;
118 
119  e->setTemplate(kCapfixPacketTid);
120  e->exportStruct(caftmpl, reinterpret_cast<uint8_t*>(&pkt));
121 }

132 int main_to_pcap(const std::string& filename) {
133 
134  // open an ipfix source
135  FileReader fr(filename);
136  pcap_t* pcap = pcap_open_dead(DLT_RAW, 65535);
137  pcap_dumper_t* dumper = pcap_dump_open(pcap, new_extension(filename, std::string("pcap")).c_str());
138  if (!dumper) {
139  std::cerr << "failed to open pcap dumper: " << pcap_geterr(pcap) << std::endl;
140  return 1;
141  }
142 
143  // create a capfix receiver around the output file
144  CapfixReceiver cr(dumper);
145 
146  // register our set receiver
147  fr.registerReceiver(&caftmpl, &cr);
148 
149  while (!didQuit()) {
150  MBuf mbuf;
151  if (!fr.receiveMessage(mbuf)) { doQuit(0); }
152  }
153 
154  // clean up (file reader closes automatically on destruction)
155  pcap_dump_close(dumper);
156  return 0;
157 }

159 int main_to_ipfix(const std::string& filename) {
160  char errbuf[PCAP_ERRBUF_SIZE];
161  pcap_t *pcap;
162 
163  // open a pcap source
164  pcap = pcap_open_offline(filename.c_str(), errbuf);
165  if (!pcap) {
166  std::cerr << "failed to open pcap file: " << errbuf << std::endl;
167  return 1;
168  }
169 
170  // make sure we like the datalink
171  datalink = pcap_datalink(pcap);
172  switch (datalink) {
173  case DLT_EN10MB:
174  case DLT_RAW:
175  break;
176  default:
177  std::cerr << "capfix only groks Ethernet and raw IP frames." << std::endl;
178  return 1;
179  break;
180  }
181 
182  // set up output
183  FileWriter fw(new_extension(filename, std::string("ipfix")), kTestDomain);
184  fw.getTemplate(kCapfixPacketTid)->mimic(caftmpl);
185  fw.getTemplate(kCapfixPacketTid)->dump(std::cerr);
186  fw.exportTemplatesForDomain();
187 
188  // runloop
189  int rv;
190  while (!didQuit()) {
191  int pcrv = pcap_dispatch(pcap, 1, exportPacket, reinterpret_cast<uint8_t *>(&fw));
192  if (pcrv == 0) {
193  // no packets, normal exit
194  rv = 0;
195  break;
196  } else if (pcrv < 0) {
197  // something broke, run away screaming
198  std::cerr << "pcap error: " << errbuf << std::endl;
199  rv = 1;
200  break;
201  }
202  }
203 
204  // clean up
205  fw.flush();
206  pcap_close(pcap);
207  return rv;
208 }

211

222  // determine what kind of file we're looking at
