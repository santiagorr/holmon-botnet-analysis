104  void _receive_msg(std::shared_ptr<const Msg>&& m, int /* index */)
105  {
106  receivedTuple++;
107  if(m->type() != m_msg_type) {
108  throw std::runtime_error("WordCounter::wrong message type");
109  }
110 
111  const WordRecord* msg_ptr = static_cast<const WordRecord *>(m.get());
112  int idx = 0;
113  assert(idx>=0 && idx < MAX_SIZE);
114  for(int i = 0; i < msg_ptr->get_word_number(); i++) {
115  receivedMessages++;
116  std::map<const std::string,unsigned long long>::iterator it;
117  if( (it = words[idx].find( msg_ptr->get_word(i))) != words[idx].end() ) {
118  it->second++;
119  } else {
120  words[idx].insert( std::pair<const std::string,unsigned long long>(msg_ptr->get_word(i),1) );
121  }
122  }
123  m_msg_recv++;
124  }
125  };
