88 void khash::compute(unsigned char *in, int len) {
89  assert(len>0);
90  unsigned msgdig[5];
91 
92  // compute first SHA hash; should be all needed most of the time;
93  // more SHA are always needed when shash*nhash>160; more SHA MAY be
94  // needed when no repetition is required... in this case this
95  // depends on the input
96  sha.Reset();
97  sha.Input(in, len);
98  bool outcome = sha.Result(msgdig);
99  assert(outcome);
100 
101  // copy sha message digest into cache; frequently not needed
102  // and can be optimized out by checking whether this copy
103  // will be needed or not;
104  unsigned char md[20];
105  copy_sha(msgdig,md);
106 
107 #ifdef _KHASH_DEBUG
108  fprintf(stderr, "\nDEBUG PRINT - original SHA1\n");
109  for(int b=0; b<5; b++) {
110  bitprint(msgdig[b], stderr);
111  fprintf(stderr,"\n");
112  }
113  fprintf(stderr, " - and its cache copy\n");
114  for(int b=1; b<=20; b++) {
115  bitprint(md[b-1], stderr);
116  if (b%4==0) fprintf(stderr,"\n");
117  }
118 #endif
119 
120  // now cycle on digest to extract khash bits from SHA digest
121  int ptr = 0;
122  int k=1;
123  while (k<=nhash) {
124  int s_blk = ptr / 32;
125  int s_bit = ptr % 32;
126  res[k] = (msgdig[s_blk] >> s_bit) & bitmask(shash);
127  int carry = s_bit + shash - 32;
128 
129 #ifdef _KHASH_DEBUG
130 fprintf(stderr, "DDD k=%d: blk=%d bit=%d carry=%d ",k,s_blk,s_bit,carry);
131 bitprint(res[k],shash,stderr);
132 fprintf(stderr, "\n");
133 #endif
134 
135  // compute next sha if needed. Note that this is done
136  // also if carry=0 for subtle reason a bit long to explain
137  if ((carry>=0)&&(s_blk==4)) {
138  sha.Reset();
139  sha.Input(md, 20);
140  bool outcome = sha.Result(msgdig);
141  assert(outcome);
142  copy_sha(msgdig,md);
143  ptr -= 160; // trick to make it fit with what follows
144  s_blk=-1; // trick to make it fit with what follows
145  }
146 
147  // now conclude computation for res, in the case
148  // computation spans over two digest unsigned blocks
149  if (carry>0) {
150  s_blk++;
151  res[k]=res[k]|((msgdig[s_blk] & bitmask(carry)) << (shash-carry));
152  }
153 
154  // more forward in the SHA digest domain
155  ptr += shash;
156  assert((ptr>=0)&&(ptr<=160));
157 
158  // finally, if norep flag is set, move to the next res
159  // only if this computed value is NOT repeated;
160  // otherwise start next iteration with the same
161  // value k (indeed k-- cancels the next k++)
162  if (norep)
163  for(int i=1; i<k; i++) if (res[i]==res[k]) k--;
164  k++;
165  }
166 
167 
168 #ifdef _KHASH_DEBUG
169  fprintf(stderr,"\nDEBUG PRINT - resulting short digests\n");
170  for(int k=1; k<=nhash; k++) {
171  bitprint(res[k],shash,stderr);
172  fprintf(stderr,"\n");
173  }
174 #endif
175 
176  }

181 // private function; it receives as input a message digest organized into
182 // a vector of 5 unsigned , and copies it into a vector of 20 unsigned char
183 //
184 void khash::copy_sha(unsigned* vin, unsigned char* vout) {
185  int blk_in=0;
186  int blk_out=0;
187  unsigned char c;
188  while(blk_in<5) {
189  c = (unsigned char)((vin[blk_in] >> (blk_out%4)*8) & 0x000000FF);
190  vout[blk_out] = c;
191  blk_out++;
192  if (blk_out%4 == 0) blk_in++;
193  }
194  assert(blk_in==5);
195  assert(blk_out == 20);
196  }
